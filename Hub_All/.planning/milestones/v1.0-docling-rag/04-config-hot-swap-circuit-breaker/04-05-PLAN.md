```go
// Reindex xóa chunks cũ + enqueue job mới với ForcedExtractor (CFG-07).
// Trả jobID = docID (job được định danh theo document).
func (s *DocumentService) Reindex(ctx context.Context, id, extractorParam, requesterID string) (string, error) {
    docUUID, err := uuid.Parse(id)
    if err != nil {
        return "", fmt.Errorf("invalid document ID")
    }

    // Validate extractor param (empty = dùng runtime config — handler resolve trước)
    switch extractorParam {
    case "docling", "native", "auto":
        // ok
    default:
        return "", fmt.Errorf("invalid extractor %q (must be docling|native|auto)", extractorParam)
    }

    doc, err := s.docRepo.FindByID(ctx, docUUID)
    if err != nil {
        return "", fmt.Errorf("find document: %w", err)
    }
    if doc == nil {
        return "", fmt.Errorf("document not found")
    }
    if doc.Status == "pending" || doc.Status == "processing" {
        return "", fmt.Errorf("document already in queue (status=%s)", doc.Status)
    }

    hub, err := s.hubRepo.FindByID(ctx, doc.HubID)
    if err != nil || hub == nil {
        return "", fmt.Errorf("hub not found for document")
    }

    // Resolve local file path — reuse logic GetFile để handle cả remote storage.
    // KHÔNG re-upload remote (per D-G.2). Chỉ đảm bảo có local copy cho extractor.
    localPath, err := s.ensureLocalCopy(ctx, doc, hub)
    if err != nil {
        return "", fmt.Errorf("ensure local copy: %w", err)
    }

    collection := hub.ChromaCollection
    if collection == "" {
        collection = "hub_" + hub.Code
    }

    // Xóa chunks cũ TRƯỚC khi enqueue (per D-G.2 chốt thiết kế).
    if err := s.docRepo.DeleteChunksByDocID(ctx, docUUID); err != nil {
        slog.Error("reindex: delete chunks Postgres failed", "doc_id", id, "error", err)
        // Tiếp tục — chunks orphan sẽ bị overwrite khi job mới chạy
    }
    if s.vecStore != nil {
        if err := s.vecStore.Delete(ctx, collection, map[string]any{"document_id": id}); err != nil {
            slog.Error("reindex: delete chunks Chroma failed", "doc_id", id, "error", err)
        }
    }

    // Reset trạng thái + clear extractor_used.
    if err := s.docRepo.UpdateStatus(ctx, docUUID, "pending", nil); err != nil {
        return "", fmt.Errorf("reset status: %w", err)
    }
    if err := s.docRepo.UpdateProgress(ctx, docUUID, 0); err != nil {
        return "", fmt.Errorf("reset progress: %w", err)
    }
    if err := s.docRepo.ClearExtractorUsed(ctx, docUUID); err != nil {
        slog.Warn("reindex: clear extractor_used failed (non-fatal)", "doc_id", id, "error", err)
    }

    // Enqueue job với ForcedExtractor.
    if s.workerMgr != nil {
        s.workerMgr.Enqueue(worker.EmbedJob{
            DocumentID:      docUUID.String(),
            DocName:         doc.Name,
            FilePath:        localPath,
            FileType:        doc.FileType,
            HubCode:         hub.Code,
            Collection:      collection,
            RequestID:       requestid.From(ctx),
            IsReindex:       true,
            ForcedExtractor: extractorParam,
        })
    } else {
        return "", fmt.Errorf("worker manager not available")
    }

    // Audit log (per D-G.2).
    if s.auditRepo != nil {
        previousExtractor := ""
        if doc.ExtractorUsed != nil {
            previousExtractor = *doc.ExtractorUsed
        }
        _ = s.auditRepo.Insert(ctx, &model.AuditLogEntry{
            Action:   "document_reindex",
            UserID:   parseUUIDPtr(requesterID),
            UserName: requesterID,
            IsAI:     false,
            Details: map[string]any{
                "document_id":        id,
                "extractor_param":    extractorParam,
                "previous_extractor": previousExtractor,
                "request_id":         requestid.From(ctx),
            },
        })
    }

    slog.Info("document reindex enqueued", "doc_id", id, "extractor", extractorParam, "requester", requesterID)
    return docUUID.String(), nil
}

// ensureLocalCopy đảm bảo có file local cho pipeline đọc.
// Nếu doc.FilePath đã là local file readable → dùng trực tiếp.
// Nếu remote (GDrive ID) → download xuống uploadDir/<hub.Code>/<docID>/<name>.
func (s *DocumentService) ensureLocalCopy(ctx context.Context, doc *model.Document, hub *model.Hub) (string, error) {
    // 1. Thử dùng doc.FilePath trực tiếp nếu là local file
    if doc.FilePath != "" {
        if _, err := os.Stat(doc.FilePath); err == nil {
            return doc.FilePath, nil
        }
    }

    // 2. Reconstruct local path canonical
    localDir := filepath.Join(s.uploadDir, hub.Code, doc.ID.String())
    localPath := filepath.Join(localDir, doc.Name)
    if _, err := os.Stat(localPath); err == nil {
        return localPath, nil
    }

    // 3. Download từ remote storage
    if s.fileStore != nil && doc.FilePath != "" {
        reader, err := s.fileStore.Download(ctx, doc.FilePath)
        if err != nil {
            return "", fmt.Errorf("remote download: %w", err)
        }
        defer reader.Close()

        if err := os.MkdirAll(localDir, 0o755); err != nil {
            return "", fmt.Errorf("mkdir local: %w", err)
        }
        out, err := os.Create(localPath)
        if err != nil {
            return "", fmt.Errorf("create local file: %w", err)
        }
        defer out.Close()
        if _, err := io.Copy(out, reader); err != nil {
            return "", fmt.Errorf("write local file: %w", err)
        }
        return localPath, nil
    }

    return "", fmt.Errorf("file not found locally and no remote storage configured")
}

// parseUUIDPtr — helper convert string sang *uuid.UUID, NULL nếu invalid.
func parseUUIDPtr(s string) *uuid.UUID {
    if u, err := uuid.Parse(s); err == nil {
        return &u
    }
    return nil
```

**3.2 `backend/internal/handler/document_handler.go`** — thêm handler:

```go
// POST /api/documents/:id/reindex
// Query param: extractor=docling|native|auto (optional; default = runtime ExtractorMode)
// Middleware: RequireAuth + RequireRole("admin") (gắn ở router)
func (h *DocumentHandler) Reindex(c *gin.Context) {
    userID, ok := middleware.GetUserID(c)
    if !ok {
        response.Unauthorized(c, "user not authenticated")
        return
    }

    id := c.Param("id")
    extractor := c.Query("extractor")
    if extractor == "" {
        // Default lấy từ runtime config — handler nhận RAGConfig getter qua DI
        // (hoặc inject vào DocumentHandler). Tạm thời mặc định "auto" nếu không có DI.
        extractor = h.defaultExtractorMode()
    }

    jobID, err := h.docService.Reindex(c.Request.Context(), id, extractor, userID.String())
    if err != nil {
        msg := err.Error()
        switch {
        case msg == "document not found":
            response.NotFound(c, msg)
        case strings.HasPrefix(msg, "invalid document ID"),
            strings.HasPrefix(msg, "invalid extractor"),
            strings.HasPrefix(msg, "document already in queue"):
            response.BadRequest(c, msg)
        default:
            slog.Error("reindex document failed", "doc_id", id, "error", err)
            response.InternalError(c, "failed to reindex document")
        }
        return
    }

    response.Accepted(c, gin.H{
        "job_id":      jobID,
        "document_id": id,
        "extractor":   extractor,
    })
}

// defaultExtractorMode trả mode runtime hiện tại (lấy từ RAGConfig getter).
// Nếu không có getter → fallback "auto".
func (h *DocumentHandler) defaultExtractorMode() string {
    if h.ragConfigGetter != nil {
        return h.ragConfigGetter()
    }
    return "auto"
}
```
