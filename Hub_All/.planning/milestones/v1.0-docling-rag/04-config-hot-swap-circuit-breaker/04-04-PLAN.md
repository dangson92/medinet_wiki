    }

func TestHealthProbe_Cache30s(t *testing.T) {
    var calls int
    ts := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        calls++
        w.WriteHeader(200)
    }))
    defer ts.Close()

    mr := miniredis.RunT(t)
    rdb := redis.NewClient(&redis.Options{Addr: mr.Addr()})
    h := NewDoclingHealthProbe(ts.URL, rdb)

    _ = h.Status(context.Background())
    callsAfterFirst := calls
    _ = h.Status(context.Background())
    _ = h.Status(context.Background())
    if calls != callsAfterFirst {
        t.Fatalf("cache miss — expected 0 new calls, got %d", calls-callsAfterFirst)

    mr.FastForward(31 * time.Second)
    _ = h.Status(context.Background())
    if calls <= callsAfterFirst {
        t.Fatalf("expected probe after cache expire")
    }
}
