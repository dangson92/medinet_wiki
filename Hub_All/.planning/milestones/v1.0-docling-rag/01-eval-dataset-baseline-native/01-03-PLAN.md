```python
"""Build 2 scanned PDF từ DOCX gốc — phục vụ eval OCR vie+eng.

Pipeline 3 bước:
  1. DOCX → PDF (có text layer) — LibreOffice headless (primary), docx2pdf (fallback 1), pandoc (fallback 2)
  2. PDF → list[PNG 150 DPI] — pdf2image.convert_from_path
  3. list[PNG] → PDF image-only — img2pdf.convert

Output:
  eval/dataset/scanned/DMD_T1-01_scanned.pdf
  eval/dataset/scanned/DMD_T1-04_scanned.pdf

Chạy: python eval/scripts/build_scanned.py [--force] [--dpi 150]
"""
from __future__ import annotations

import argparse
import logging
import shutil
import subprocess
import sys
import tempfile
from pathlib import Path

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
logger = logging.getLogger(__name__)

REPO_ROOT = Path(__file__).resolve().parents[2]
SOURCES_DIR = REPO_ROOT / "eval" / "dataset" / "sources"
OUTPUT_DIR = REPO_ROOT / "eval" / "dataset" / "scanned"

# 2 file mục tiêu, đã chốt ở CONTEXT.md mục B
TARGETS = [
    ("DMD_T1-01_DinhVi_TrungTam_v1.docx", "DMD_T1-01_scanned.pdf"),
    ("DMD_T1-04_FAQ_ThuongHieu_v1.docx", "DMD_T1-04_scanned.pdf"),
]


# ─── Step 1: DOCX → PDF (text layer) — fallback chain ─────────────────────
def convert_docx_to_pdf(docx: Path, out_dir: Path) -> Path:
    """Thử LibreOffice → docx2pdf → pandoc, trả Path PDF có text layer."""
    out_pdf = out_dir / (docx.stem + ".pdf")

    # Primary: LibreOffice headless (cross-platform)
    soffice = shutil.which("soffice") or shutil.which("libreoffice")
    if soffice:
        logger.info("Step 1 [LibreOffice]: %s → %s", docx.name, out_pdf.name)
        try:
            subprocess.run(
                [soffice, "--headless", "--convert-to", "pdf", "--outdir", str(out_dir), str(docx)],
                check=True, timeout=120, capture_output=True,
            )
            if out_pdf.exists():
                return out_pdf
        except (subprocess.CalledProcessError, subprocess.TimeoutExpired) as e:
            logger.warning("LibreOffice fail (%s), thử fallback...", e)

    # Fallback 1: docx2pdf (Windows + MS Word, hoặc macOS)
    try:
        from docx2pdf import convert  # type: ignore
        logger.info("Step 1 [docx2pdf]: %s → %s", docx.name, out_pdf.name)
        convert(str(docx), str(out_pdf))
        if out_pdf.exists():
            return out_pdf
    except ImportError:
        logger.debug("docx2pdf không cài — skip")
    except Exception as e:
        logger.warning("docx2pdf fail (%s), thử fallback tiếp...", e)

    # Fallback 2: pandoc
    pandoc = shutil.which("pandoc")
    if pandoc:
        logger.info("Step 1 [pandoc]: %s → %s", docx.name, out_pdf.name)
        try:
            subprocess.run(
                [pandoc, str(docx), "-o", str(out_pdf), "--pdf-engine=wkhtmltopdf"],
                check=True, timeout=120, capture_output=True,
            )
            if out_pdf.exists():
                return out_pdf
        except (subprocess.CalledProcessError, subprocess.TimeoutExpired) as e:
            logger.warning("pandoc fail (%s)", e)

    raise RuntimeError(
        f"Không convert được {docx.name} → PDF. "
        "Cài LibreOffice (soffice trong PATH) HOẶC `pip install docx2pdf` HOẶC pandoc + wkhtmltopdf."
    )


# ─── Step 2: PDF → list[PNG] qua pdf2image ────────────────────────────────
def pdf_to_images(pdf: Path, dpi: int) -> list[bytes]:
    """Render mỗi page PDF thành PNG bytes (in-memory)."""
    from io import BytesIO
    from pdf2image import convert_from_path  # type: ignore

    logger.info("Step 2 [pdf2image]: %s @ %d DPI", pdf.name, dpi)
    images = convert_from_path(str(pdf), dpi=dpi)
    png_bytes_list: list[bytes] = []
    for img in images:
        buf = BytesIO()
        img.save(buf, format="PNG")
        png_bytes_list.append(buf.getvalue())
    logger.info("  → %d page rasterized", len(png_bytes_list))
    return png_bytes_list


# ─── Step 3: list[PNG] → PDF image-only ──────────────────────────────────
def images_to_image_pdf(png_list: list[bytes], out_pdf: Path) -> None:
    """Ghép list PNG bytes thành PDF image-only (no text layer)."""
    import img2pdf  # type: ignore

    logger.info("Step 3 [img2pdf]: %d page → %s", len(png_list), out_pdf.name)
    out_pdf.parent.mkdir(parents=True, exist_ok=True)
    pdf_bytes = img2pdf.convert(png_list)
    out_pdf.write_bytes(pdf_bytes)


# ─── Verify: PDF không có text layer ─────────────────────────────────────
def verify_no_text_layer(pdf: Path) -> None:
    """Sanity check: dùng pdftotext nếu available, expect output rỗng."""
    pdftotext = shutil.which("pdftotext")
    if not pdftotext:
        logger.info("pdftotext không cài → skip verify (manual check khuyến nghị)")
        return
    result = subprocess.run(
        [pdftotext, str(pdf), "-"], capture_output=True, text=True, timeout=30
    )
    text = result.stdout.strip()
    if text:
        logger.warning(
            "⚠ %s vẫn có text layer (%d chars) — có thể img2pdf bug hoặc step 2 sai DPI",
            pdf.name, len(text),
        )
    else:
        logger.info("✓ %s không có text layer (đúng yêu cầu scanned)", pdf.name)


# ─── Main ────────────────────────────────────────────────────────────────
def build_one(docx_name: str, out_name: str, dpi: int, force: bool) -> None:
    docx_path = SOURCES_DIR / docx_name
    out_path = OUTPUT_DIR / out_name

    if not docx_path.exists():
        raise FileNotFoundError(f"Không tìm thấy {docx_path}")

    if out_path.exists() and not force:
        logger.info("✓ %s đã tồn tại — skip (dùng --force để rebuild)", out_path.name)
        return

    with tempfile.TemporaryDirectory(prefix="eval_scanned_") as tmp:
        tmp_dir = Path(tmp)
        text_pdf = convert_docx_to_pdf(docx_path, tmp_dir)
        png_list = pdf_to_images(text_pdf, dpi)
        images_to_image_pdf(png_list, out_path)
        verify_no_text_layer(out_path)


def main() -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument("--force", action="store_true", help="Rebuild kể cả khi output đã tồn tại")
    parser.add_argument("--dpi", type=int, default=150, help="DPI khi rasterize (default 150)")
    args = parser.parse_args()

    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
    for docx_name, out_name in TARGETS:
        try:
            build_one(docx_name, out_name, args.dpi, args.force)
        except Exception as e:
            logger.error("Build %s thất bại: %s", out_name, e)
            return 1

    logger.info("Build scanned hoàn tất: %d file ở %s", len(TARGETS), OUTPUT_DIR)
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
