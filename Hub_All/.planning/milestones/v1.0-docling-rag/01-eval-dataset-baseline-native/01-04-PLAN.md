"""Extract heading path từ DOCX styles — port từ backend/internal/rag/extractor/docx.go:46-95.

Thuật toán:
1. Mở DOCX (zip) → đọc word/styles.xml.
2. Với mỗi <w:style w:styleId="X">, tìm <w:outlineLvl w:val="N"/> → headingLevel = N+1.
3. Fallback hard-coded: styleId "Heading1".."Heading6" → level 1..6.
4. Đọc word/document.xml, duyệt <w:p>, lookup pStyle → nếu là heading thì push lên stack.
5. Output `eval/dataset/headings.json`: {doc_id: [heading_path_1, heading_path_2, ...]}.

PDF gốc + 2 scanned PDF: heading manual (extractor Go unreliable cho PDF; scanned PDF chưa OCR).
  - PDF gốc `tri_thuc_chinh_tri.pdf` → manual entry trong code (constant MANUAL_PDF_HEADINGS).
  - 2 scanned PDF → copy từ DOCX gốc (vì cùng nguồn).

Chạy: python eval/scripts/extract_headings.py
"""
from __future__ import annotations

import json
import logging
import sys
import zipfile
from pathlib import Path
from xml.etree import ElementTree as ET

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
logger = logging.getLogger(__name__)

REPO_ROOT = Path(__file__).resolve().parents[2]
SOURCES_DIR = REPO_ROOT / "eval" / "dataset" / "sources"
SCANNED_DIR = REPO_ROOT / "eval" / "dataset" / "scanned"
OUTPUT_JSON = REPO_ROOT / "eval" / "dataset" / "headings.json"

NS_W = "{http://schemas.openxmlformats.org/wordprocessingml/2006/main}"

# Mapping scanned ↔ DOCX gốc (CONTEXT.md mục B)
SCANNED_SOURCE_MAP = {
    "DMD_T1-01_scanned.pdf": "DMD_T1-01_DinhVi_TrungTam_v1.docx",
    "DMD_T1-04_scanned.pdf": "DMD_T1-04_FAQ_ThuongHieu_v1.docx",
}

# Manual heading cho PDF gốc — extractor Go heading detection cho PDF unreliable.
# REVISION 1 (W4): Phải có ≥ 1 entry. Executor (LLM Claude) phải mở PDF qua pypdf,
# đọc TOC + 5–10 trang đầu, liệt kê heading vàng draft. User review trong Plan 05 checkpoint.
MANUAL_PDF_HEADINGS: dict[str, list[str]] = {
    "tri_thuc_chinh_tri.pdf": [
        # TODO(executor-task-2): mở PDF qua pypdf trong Task 2 dưới đây,
        # đọc 5-10 trang đầu + TOC nếu có, liệt kê heading thật tại đây.
        # Format: "<lvl1>", "<lvl1> > <lvl2>", "<lvl1> > <lvl2> > <lvl3>"
        # Ví dụ:
        #   "Phần I: Lý luận chung",
        #   "Phần I: Lý luận chung > Chương 1: Khái niệm",
        #   "Phần I: Lý luận chung > Chương 1: Khái niệm > Mục A",
        # KHÔNG để list rỗng — Plan 05 query bonus có thể trỏ vào file này.
    ],
}


def parse_docx_styles(docx_path: Path) -> dict[str, int]:
    """Trả về map {styleId: headingLevel}. headingLevel=0 nghĩa không phải heading.

    Port từ docx.go:67-83.
    """
    styles: dict[str, int] = {}
    with zipfile.ZipFile(docx_path) as z:
        try:
            with z.open("word/styles.xml") as f:
                tree = ET.parse(f)
        except KeyError:
            logger.warning("%s không có word/styles.xml", docx_path.name)
            tree = None

    if tree is not None:
        for style_el in tree.findall(f"{NS_W}style"):
            style_id = style_el.get(f"{NS_W}styleId")
            if not style_id:
                continue
            outline = style_el.find(f".//{NS_W}outlineLvl")
            if outline is not None:
                val = outline.get(f"{NS_W}val")
                if val is not None and val.isdigit():
                    # Go: headingLevel = N + 1 (0-based → 1-based)
                    styles[style_id] = int(val) + 1

    # Fallback hard-coded (docx.go:88-93)
    for i in range(1, 7):
        styles.setdefault(f"Heading{i}", i)
    return styles


def extract_heading_paths(docx_path: Path) -> list[str]:
    """Duyệt body paragraphs, build heading path nối bằng " > "."""
    styles = parse_docx_styles(docx_path)
    paths: list[str] = []
    stack: list[str] = []  # stack[i] = title của heading level (i+1)

    with zipfile.ZipFile(docx_path) as z:
        with z.open("word/document.xml") as f:
            tree = ET.parse(f)

    body = tree.getroot().find(f"{NS_W}body")
    if body is None:
        logger.warning("%s không có <w:body>", docx_path.name)
        return paths

    for p in body.findall(f"{NS_W}p"):
        p_style = p.find(f".//{NS_W}pStyle")
        if p_style is None:
            continue
        style_id = p_style.get(f"{NS_W}val")
        level = styles.get(style_id, 0) if style_id else 0
        if level <= 0 or level > 6:
            continue

        # Lấy text từ tất cả run trong paragraph
        text_runs = [t.text or "" for t in p.findall(f".//{NS_W}t")]
        title = "".join(text_runs).strip()
        if not title:
            continue

        # Truncate stack về đúng độ sâu (level-1) rồi push title mới
        del stack[level - 1 :]
        stack.append(title)
        paths.append(" > ".join(stack))

    return paths


def main() -> int:
    if not SOURCES_DIR.exists():
        logger.error("Không tìm thấy %s — chạy Plan 01 trước", SOURCES_DIR)
        return 1

    headings: dict[str, list[str]] = {}

    # 1. Extract auto từ tất cả DOCX trong sources/
    for docx in sorted(SOURCES_DIR.glob("*.docx")):
        try:
            paths = extract_heading_paths(docx)
            headings[docx.name] = paths
            logger.info("✓ %s: %d heading", docx.name, len(paths))
        except Exception as e:
            logger.error("✗ %s: %s", docx.name, e)
            return 2

    # 2. Manual cho PDF gốc — REVISION 1 (W4): bắt buộc có ≥ 1 entry
    for pdf_name, manual_paths in MANUAL_PDF_HEADINGS.items():
        pdf_path = SOURCES_DIR / pdf_name
        if pdf_path.exists():
            headings[pdf_name] = manual_paths
            if not manual_paths:
                logger.error(
                    "✗ %s có entry manual nhưng list rỗng — Task 2 (Plan 04) phải fill MANUAL_PDF_HEADINGS trước khi chạy lại.",
                    pdf_name,
                )
                return 3
            logger.info("✓ %s (manual): %d heading", pdf_name, len(manual_paths))

    # 3. Scanned PDF — copy từ DOCX gốc tương ứng
    for scanned_name, source_docx in SCANNED_SOURCE_MAP.items():
        scanned_path = SCANNED_DIR / scanned_name
        if not scanned_path.exists():
            logger.warning("⚠ %s chưa tồn tại — chạy Plan 03 trước", scanned_name)
            continue
        if source_docx in headings:
            headings[scanned_name] = headings[source_docx]
            logger.info(
                "✓ %s (copy từ %s): %d heading",
                scanned_name, source_docx, len(headings[scanned_name]),
            )

    # 4. Write JSON
    OUTPUT_JSON.parent.mkdir(parents=True, exist_ok=True)
    OUTPUT_JSON.write_text(
        json.dumps(headings, ensure_ascii=False, indent=2, sort_keys=True),
        encoding="utf-8",
    )
    logger.info(
        "Đã ghi %s — %d doc, tổng %d heading path",
        OUTPUT_JSON, len(headings), sum(len(v) for v in headings.values()),
    )
    return 0


if __name__ == "__main__":
    sys.exit(main())
