```python
"""Reset eval state trước mỗi baseline run.

Chạy: python eval/scripts/cleanup.py [--keep-uploads] [--keep-audit]

Bước:
1. Lấy eval_hub_id từ code='eval' (fail loud nếu chưa seed).
2. Kiểm tra không có document `pending`/`processing` (race với worker pool Go).
3. DELETE FROM documents WHERE hub_id=<eval_id> → cascade xóa document_chunks.
4. DELETE ChromaDB collection `medinet_eval` qua v2 API → re-create ngay (get_or_create=true).
5. shutil.rmtree backend/uploads/eval/ — guard chặn EVAL_HUB_CODE rỗng/dangerous (skip nếu --keep-uploads).
6. (Optional) DELETE FROM audit_logs WHERE hub_id=<eval_id> (skip nếu --keep-audit).
"""
from __future__ import annotations

import argparse
import logging
import os
import shutil
import sys
from pathlib import Path

import httpx
import psycopg
from dotenv import load_dotenv

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s",
)
logger = logging.getLogger(__name__)

REPO_ROOT = Path(__file__).resolve().parents[2]
load_dotenv(REPO_ROOT / "eval" / ".env")

# Config từ env
DB_HOST = os.getenv("DB_HOST", "localhost")
DB_PORT = int(os.getenv("DB_PORT", "5432"))
DB_NAME = os.getenv("DB_NAME", "medinet_central")
DB_USER = os.getenv("DB_USER", "medinet")
DB_PASSWORD = os.getenv("DB_PASSWORD", "")
CHROMA_URL = os.getenv("CHROMA_URL", "http://localhost:8000")
CHROMA_TOKEN = os.getenv("CHROMA_TOKEN", "")
EVAL_HUB_CODE = os.getenv("EVAL_HUB_CODE", "eval")
EVAL_COLLECTION = os.getenv("EVAL_COLLECTION", "medinet_eval")

V2_BASE = "/api/v2/tenants/default_tenant/databases/default_database"

# Guard chống xóa nhầm thư mục: tập giá trị "dangerous" cho EVAL_HUB_CODE
DANGEROUS_HUB_CODES = {"", ".", "/", "*", "..", "\\"}


def chroma_headers() -> dict[str, str]:
    headers = {"Content-Type": "application/json"}
    if CHROMA_TOKEN:
        headers["Authorization"] = f"Bearer {CHROMA_TOKEN}"
    return headers


def get_eval_hub_id(conn: psycopg.Connection) -> str:
    """Trả về UUID của hub `eval`. Fail loud nếu chưa seed."""
    with conn.cursor() as cur:
        cur.execute("SELECT id FROM hubs WHERE code = %s", (EVAL_HUB_CODE,))
        row = cur.fetchone()
        if not row:
            logger.error(
                "Hub code='%s' chưa tồn tại. Chạy seed trước:\n"
                "  psql -h %s -U %s -d %s -f eval/scripts/seed_hub.sql",
                EVAL_HUB_CODE, DB_HOST, DB_USER, DB_NAME,
            )
            sys.exit(2)
        return str(row[0])


def assert_no_active_jobs(conn: psycopg.Connection, hub_id: str) -> None:
    """Race-safety: không xóa khi worker còn process job pending/processing."""
    with conn.cursor() as cur:
        cur.execute(
            "SELECT COUNT(*) FROM documents WHERE hub_id = %s AND status IN ('pending','processing')",
            (hub_id,),
        )
        count = cur.fetchone()[0]
        if count > 0:
            logger.error(
                "Có %d document đang pending/processing — chờ worker drain xong rồi chạy lại.",
                count,
            )
            sys.exit(3)


def cleanup_documents(conn: psycopg.Connection, hub_id: str) -> int:
    """DELETE documents — cascade xóa document_chunks qua FK."""
    with conn.cursor() as cur:
        cur.execute("DELETE FROM documents WHERE hub_id = %s", (hub_id,))
        deleted = cur.rowcount
        conn.commit()
        logger.info("Đã xóa %d documents (chunks tự cascade)", deleted)
        return deleted


def cleanup_audit_logs(conn: psycopg.Connection, hub_id: str) -> int:
    """DELETE audit_logs — partition theo timestamp tự apply trên all partitions."""
    with conn.cursor() as cur:
        cur.execute("DELETE FROM audit_logs WHERE hub_id = %s", (hub_id,))
        deleted = cur.rowcount
        conn.commit()
        logger.info("Đã xóa %d audit_logs", deleted)
        return deleted


def cleanup_chroma_collection() -> None:
    """DELETE rồi POST get_or_create để re-create ngay (không cần restart backend Go)."""
    headers = chroma_headers()
    with httpx.Client(timeout=30, headers=headers) as client:
        # DELETE — 404 OK (collection có thể chưa tồn tại)
        del_url = f"{CHROMA_URL}{V2_BASE}/collections/{EVAL_COLLECTION}"
        try:
            r = client.delete(del_url)
            if r.status_code in (200, 204, 404):
                logger.info("ChromaDB DELETE %s → %d", EVAL_COLLECTION, r.status_code)
            else:
                logger.warning("ChromaDB DELETE %s → %d %s", EVAL_COLLECTION, r.status_code, r.text)
        except httpx.RequestError as e:
            logger.error("ChromaDB DELETE lỗi network: %s", e)
            sys.exit(4)

        # POST get_or_create
        create_url = f"{CHROMA_URL}{V2_BASE}/collections"
        body = {
            "name": EVAL_COLLECTION,
            "metadata": {"hnsw:space": "cosine"},
            "get_or_create": True,
        }
        r = client.post(create_url, json=body)
        if r.status_code not in (200, 201):
            logger.error("ChromaDB CREATE %s thất bại: %d %s", EVAL_COLLECTION, r.status_code, r.text)
            sys.exit(5)
        logger.info("ChromaDB CREATE %s → %d", EVAL_COLLECTION, r.status_code)


def cleanup_uploads_dir() -> None:
    """Xóa backend/uploads/eval/ (file local).

    REVISION 1 (I1): Guard 3 lớp chống xóa nhầm:
      1. EVAL_HUB_CODE không được rỗng / "." / "/" / "*" / ".." / "\\".
      2. Path target phải là subdir thật của backend/uploads/.
      3. Path string phải chứa "eval" — sanity check cuối.
    """
    # Guard 1: env value
    if not EVAL_HUB_CODE or EVAL_HUB_CODE in DANGEROUS_HUB_CODES:
        logger.error(
            "Refusing rmtree: EVAL_HUB_CODE='%s' rỗng/dangerous — KHÔNG xóa thư mục root backend/uploads/.",
            EVAL_HUB_CODE,
        )
        sys.exit(99)

    uploads_root = REPO_ROOT / "backend" / "uploads"
    target = uploads_root / EVAL_HUB_CODE

    # Guard 2: target là subdir của uploads_root + tồn tại + là dir
    if not target.exists():
        logger.info("Thư mục %s không tồn tại — skip", target)
        return
    if not target.is_dir():
        logger.error("%s không phải là directory — refuse rmtree", target)
        sys.exit(99)

    # Guard 3: sanity — path string phải chứa "eval"
    if "eval" not in str(target):
        logger.error("Sanity fail: path %s không chứa 'eval' — refuse rmtree", target)
        sys.exit(99)

    shutil.rmtree(target)
    logger.info("Đã xóa thư mục %s", target)


def main() -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument("--keep-uploads", action="store_true", help="Giữ file local")
    parser.add_argument("--keep-audit", action="store_true", help="Giữ audit logs")
    args = parser.parse_args()

    dsn = f"host={DB_HOST} port={DB_PORT} dbname={DB_NAME} user={DB_USER} password={DB_PASSWORD}"
    with psycopg.connect(dsn) as conn:
        hub_id = get_eval_hub_id(conn)
        logger.info("Eval hub_id = %s", hub_id)
        assert_no_active_jobs(conn, hub_id)
        cleanup_documents(conn, hub_id)
        if not args.keep_audit:
            cleanup_audit_logs(conn, hub_id)

    cleanup_chroma_collection()
    if not args.keep_uploads:
        cleanup_uploads_dir()

    logger.info("Cleanup hoàn tất.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
