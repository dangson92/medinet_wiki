```python
"""Serialize Docling output → response dict đúng schema DSVC-02.

Schema chốt trong CONTEXT.md mục "Schema response /v1/process".
Plan 06 (API endpoint) dùng và Phase 3 (Go adapter) deserialize — contract ĐÓNG BĂNG.

Trách nhiệm:
1. build_doc_meta(doc, filename, ocr_used) → dict doc_meta.
2. serialize_chunks(doc, chunks, tokenizer) → list dict đúng schema.
3. Detect is_table + extract table_html (EXTRACT-03).
4. Inject caption figure dạng ![<caption>](#fig-N) vào text (EXTRACT-04).
5. Compute page_start/end + bbox (union) từ chunk.meta.doc_items[].prov.
"""

from __future__ import annotations

from pathlib import Path
from typing import Any, TypedDict

import structlog
import tiktoken
from docling_core.transforms.chunker import DocChunk
from docling_core.types.doc import DoclingDocument

logger = structlog.get_logger(__name__)


class DocMetaDict(TypedDict):
    filename: str
    file_type: str
    page_count: int
    language_detected: str | None
    ocr_used: bool


class ChunkDict(TypedDict):
    chunk_index: int
    text: str
    headers: list[str]
    caption: str | None
    page_start: int | None
    page_end: int | None
    is_table: bool
    table_html: str | None
    bbox: list[float] | None
    token_count: int


def _file_type_from_ext(filename: str) -> str:
    ext = Path(filename).suffix.lower().lstrip(".")
    return ext or "unknown"


def build_doc_meta(
    doc: DoclingDocument,
    filename: str,
    ocr_used: bool,
    language_detected: str | None = None,
) -> DocMetaDict:
    """Build doc_meta dict đúng schema DSVC-02."""
    page_count = len(doc.pages) if hasattr(doc, "pages") and doc.pages else 0
    return DocMetaDict(
        filename=filename,
        file_type=_file_type_from_ext(filename),
        page_count=page_count,
        language_detected=language_detected,  # Docling không expose detect — leave None M1
        ocr_used=ocr_used,
    )


def _extract_provenance(chunk: DocChunk) -> tuple[int | None, int | None, list[float] | None]:
    """Lấy page_start, page_end, bbox (union) từ chunk.meta.doc_items[].prov."""
    pages: list[int] = []
    bboxes: list[tuple[float, float, float, float]] = []

    items = getattr(chunk.meta, "doc_items", None) or []
    for item in items:
        prov_list = getattr(item, "prov", None) or []
        for prov in prov_list:
            page_no = getattr(prov, "page_no", None)
            if page_no is not None:
                pages.append(int(page_no))
            bbox = getattr(prov, "bbox", None)
            if bbox is not None:
                # bbox có shape l/t/r/b hoặc x0/y0/x1/y1 tùy version
                coords = (
                    float(getattr(bbox, "l", getattr(bbox, "x0", 0.0))),
                    float(getattr(bbox, "t", getattr(bbox, "y0", 0.0))),
                    float(getattr(bbox, "r", getattr(bbox, "x1", 0.0))),
                    float(getattr(bbox, "b", getattr(bbox, "y1", 0.0))),
                )
                bboxes.append(coords)

    page_start = min(pages) if pages else None
    page_end = max(pages) if pages else None

    if bboxes:
        # Union bbox: min(l,t), max(r,b) — đủ cho citation rendering ở UI
        bbox_union = [
            min(b[0] for b in bboxes),
            min(b[1] for b in bboxes),
            max(b[2] for b in bboxes),
            max(b[3] for b in bboxes),
        ]
    else:
        bbox_union = None

    return page_start, page_end, bbox_union


def _detect_table_html(doc: DoclingDocument, chunk: DocChunk) -> tuple[bool, str | None]:
    """Detect chunk có chứa table → trả (is_table, table_html).

    Logic: duyệt chunk.meta.doc_items, nếu có item label TABLE thì lấy
    item.export_to_html() (Docling 2.x DocItem TableItem có method này).
    """
    items = getattr(chunk.meta, "doc_items", None) or []
    for item in items:
        label_str = str(getattr(item, "label", "")).lower()
        if "table" in label_str:
            try:
                # TableItem có export_to_html() trong Docling 2.x
                html = item.export_to_html(doc=doc) if hasattr(item, "export_to_html") else None
                if html:
                    return True, html
            except Exception as exc:
                logger.warning("table_html_extract_fail", error=str(exc))
                return True, None
    return False, None


def _extract_caption(chunk: DocChunk) -> str | None:
    """Lấy caption đầu tiên trong chunk.meta.captions (EXTRACT-04)."""
    captions = getattr(chunk.meta, "captions", None) or []
    if captions:
        return str(captions[0])
    return None


def _inject_figure_caption_marker(text: str, chunk: DocChunk, fig_index: int) -> str:
    """Inject ![caption](#fig-N) vào text nếu chunk chứa figure có caption.

    EXTRACT-04: KHÔNG insert binary image — chỉ caption marker.
    """
    items = getattr(chunk.meta, "doc_items", None) or []
    has_picture = any("picture" in str(getattr(it, "label", "")).lower() for it in items)
    if not has_picture:
        return text

    caption = _extract_caption(chunk) or "figure"
    marker = f"![{caption}](#fig-{fig_index})"
    # Append marker đầu chunk text — Markdown sẽ render placeholder anchor
    return f"{marker}\n\n{text}"


def serialize_chunks(
    doc: DoclingDocument,
    chunks: list[DocChunk],
    tokenizer_name: str = "cl100k_base",
) -> list[ChunkDict]:
    """Map list DocChunk → list ChunkDict đúng schema DSVC-02."""
    try:
        encoding = tiktoken.get_encoding(tokenizer_name)
    except Exception:
        # Tokenizer fail → token_count fallback len(text)//4 (rough estimate)
        encoding = None
        logger.warning("tokenizer_load_fail_fallback_estimate", name=tokenizer_name)

    out: list[ChunkDict] = []
    fig_counter = 0

    for idx, ch in enumerate(chunks):
        page_start, page_end, bbox = _extract_provenance(ch)
        is_table, table_html = _detect_table_html(doc, ch)
        caption = _extract_caption(ch)

        text = ch.text or ""
        # Inject figure marker nếu chunk chứa picture
        items = getattr(ch.meta, "doc_items", None) or []
        if any("picture" in str(getattr(it, "label", "")).lower() for it in items):
            fig_counter += 1
            text = _inject_figure_caption_marker(text, ch, fig_counter)

        if encoding is not None:
            token_count = len(encoding.encode(text))
        else:
            token_count = max(1, len(text) // 4)

        headers = list(getattr(ch.meta, "headings", None) or [])

        out.append(
            ChunkDict(
                chunk_index=idx,
                text=text,
                headers=headers,
                caption=caption,
                page_start=page_start,
                page_end=page_end,
                is_table=is_table,
                table_html=table_html,
                bbox=bbox,
                token_count=token_count,
            )
        )

    return out
```
