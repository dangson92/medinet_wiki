Pattern 307 redirect (FastAPI built-in):

```python
from fastapi.responses import RedirectResponse

@router.post("/login")
async def login(request: Request, ...) -> JSONResponse | RedirectResponse:
    settings = get_settings()
    if settings.hub_name != "central":
        # Hub con — redirect 307 (preserve method + body)
        return RedirectResponse(
            url=f"{settings.central_url}/api/auth/login",
            status_code=307,
        )
    # Central — handle local M2 path
    try:
        result = await service.login(req)
    except AuthError as e: ...
    return resp.ok(data=result.model_dump(mode="json"))
```

**Quan trọng — 307 vs 308 vs 302:**

- **307 (Temporary Redirect):** Preserve HTTP method (POST → POST). Browser auto-follow ✓. RFC 7231 standard cho method-preserving temp redirect.
- **308 (Permanent Redirect):** Tương tự 307 nhưng permanent (cache aggressive). KHÔNG dùng — SSO redirect temporary by design (operator switch central URL dễ hơn).
- **302 (Found):** Lịch sử dùng để redirect POST→GET (browser quirk — KHÔNG preserve method). KHÔNG dùng — Phase 5 frontend POST login body sẽ mất.

Plan 03-04 dùng 307 — preserve POST + body (LoginRequest JSON) khi browser auto-follow tới central.

```python
from fastapi.responses import JSONResponse, RedirectResponse

from app.config import get_settings


@router.post("/login")
async def login(
    request: Request,
    req: LoginRequest,
    service: AuthService = Depends(get_auth_service),  # noqa: B008
) -> JSONResponse | RedirectResponse:
    """Login endpoint — hub con redirect tới central; central handle local.

    Phase 3 Plan 03-04 SSO-02 (D-V3-Phase3-G): Hub con KHÔNG sinh JWT/refresh
    token — 100% session lifecycle ở central. Hub con trả 307 Location: central
    để browser auto-follow + preserve POST method + body.
    """
    settings = get_settings()
    if settings.hub_name != "central":
        # Defensive — Settings validator Plan 03-04 Task 1 đã enforce central_url
        # required ở hub con; bao thêm runtime fallback tránh URL rỗng.
        if not settings.central_url:
            return resp.error(
                status_code=503,
                code="CENTRAL_URL_UNAVAILABLE",
                message=(
                    "Hub con KHÔNG configure CENTRAL_URL env var — "
                    "vui lòng liên hệ admin."
                ),
            )
        # 307 preserve POST method + body (RFC 7231). Browser auto-follow.
        return RedirectResponse(
            url=f"{settings.central_url}/api/auth/login",
            status_code=307,
            headers={
                "X-SSO-Redirect-Reason": "hub_con_no_local_auth",
                "X-SSO-Original-Hub": settings.hub_name,
            },
        )

    # Central path — M2 logic carry forward
    try:
        result = await service.login(req)
    except AuthError as e:
        return _auth_error_to_response(e)
    return resp.ok(data=result.model_dump(mode="json"))
```

```python
    """Refresh endpoint — hub con redirect tới central (D-V3-Phase3-C/G)."""
    settings = get_settings()
    if settings.hub_name != "central":
        if not settings.central_url:
            return resp.error(
                status_code=503,
                code="CENTRAL_URL_UNAVAILABLE",
                message=(
                    "Hub con KHÔNG configure CENTRAL_URL env var — "
                    "vui lòng liên hệ admin."
                ),
            )
        return RedirectResponse(
            url=f"{settings.central_url}/api/auth/refresh",
            status_code=307,
            headers={
                "X-SSO-Redirect-Reason": "hub_con_no_local_refresh",
                "X-SSO-Original-Hub": settings.hub_name,
            },
        )

    # Central path — M2 logic carry forward
    try:
        result = await service.refresh(req)
    except AuthError as e:
        return _auth_error_to_response(e)
    return resp.ok(data=result.model_dump(mode="json"))
```
