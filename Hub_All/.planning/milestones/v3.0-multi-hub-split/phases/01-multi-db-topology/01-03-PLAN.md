```python
# === DSN injection runtime — support -x hub=<name> override ===
#
# Modes (Phase 1 TOPO-02):
# 1. `alembic upgrade head`              → dùng Settings.database_url (M2 backward-compat)
# 2. `alembic -x hub=<name> upgrade head` → override DSN per-hub (Plan 03 + 04 callers)
#
# resolve_database_url (Plan 02 ship) thay segment `medinet_central` →
# `medinet_hub_<name>` trong DSN. Caller phải set DATABASE_URL trỏ medinet_central
# ở .env (base DSN) — env.py tự thay segment theo -x hub.
#
# Cross-hub roundtrip (W8): caller có thể chạy với DATABASE_URL=...medinet_hub_yte
# (Settings.hub_name=yte) + pass -x hub=duoc → env un-resolve về central rồi
# resolve sang duoc. Support pattern này để dev test cross-hub migration mà
# KHÔNG cần restart server với env khác.

from app.config import get_settings, resolve_database_url

_VALID_HUBS = {"central", "yte", "duoc", "hcns"}


def parse_hub_x_arg(x_args: list[str]) -> str | None:
    """Parse `-x hub=<name>` từ list x_arguments — trả hub name hoặc None.

    `context.get_x_argument(as_dictionary=False)` trả list ["key=value", ...].

    Raises:
        ValueError: nếu hub=<name> không thuộc 4 hub hợp lệ.
    """
    for arg in x_args:
        if not arg.startswith("hub="):
            continue
        hub = arg.split("=", 1)[1].strip()
        if hub not in _VALID_HUBS:
            raise ValueError(
                f"alembic -x hub=<name>: '{hub}' không hợp lệ. "
                f"Hợp lệ: {sorted(_VALID_HUBS)}."
            )
        return hub
    return None


def resolve_env_database_url(
    base_dsn: str, hub_arg: str | None, default_hub: str
) -> str:
    """Resolve DSN runtime — ưu tiên hub_arg, fallback default_hub (Settings.hub_name).

    Args:
        base_dsn: Settings.database_url (đã validated khớp default_hub theo Plan 02).
        hub_arg: từ `alembic -x hub=<name>` hoặc None.
        default_hub: Settings.hub_name fallback.

    Returns:
        DSN đã resolved cho target hub.

    Supported scenarios:
    - target=central + base_dsn trỏ central: return base_dsn (no rewriting).
    - target=<hub> + base_dsn trỏ central: resolve_database_url(base_dsn, hub).
    - target=<hub> + base_dsn trỏ medinet_hub_<other>: un-resolve về central trước,
      rồi resolve sang target (W8 — cross-hub yte→duoc roundtrip).
    - target=central + base_dsn trỏ medinet_hub_<other>: un-resolve về central.

    Raises:
        ValueError: base_dsn không trỏ medinet_central HOẶC medinet_hub_* (input sai).
    """
    target_hub = hub_arg if hub_arg is not None else default_hub
    if target_hub == "central":
        # Verify base_dsn — nếu đã central return nguyên; nếu trỏ hub_* un-resolve.
        path_part = base_dsn.split("?", 1)[0].rstrip("/")
        if path_part.endswith("/medinet_central"):
            return base_dsn
        # Un-resolve hub_* về central
        last_seg = path_part.rsplit("/", 1)[-1]
        if last_seg.startswith("medinet_hub_"):
            query_suffix = "" if "?" not in base_dsn else "?" + base_dsn.split("?", 1)[1]
            return (
                path_part[: -len(last_seg)] + "medinet_central" + query_suffix
            )
        raise ValueError(
            f"base_dsn không trỏ medinet_central hoặc medinet_hub_*: {base_dsn!r}"
        )
    # target_hub != central → base_dsn phải trỏ central để resolve_database_url thay segment
    base_dsn_for_resolve = base_dsn
    if not base_dsn.split("?", 1)[0].rstrip("/").endswith("/medinet_central"):
        # Caller dùng `alembic -x hub=duoc` với DATABASE_URL=...medinet_hub_yte
        # (Settings.hub_name=yte). Un-resolve về central trước (W8 cross-hub):
        path_part = base_dsn.split("?", 1)[0]
        query_suffix = "" if "?" not in base_dsn else "?" + base_dsn.split("?", 1)[1]
        # Find segment `medinet_hub_<something>` ở cuối path
        last_seg = path_part.rsplit("/", 1)[-1]
        if last_seg.startswith("medinet_hub_"):
            base_dsn_for_resolve = (
                path_part[: -len(last_seg)] + "medinet_central" + query_suffix
            )
        else:
            raise ValueError(
                f"base_dsn không trỏ medinet_central hoặc medinet_hub_*: {base_dsn!r}"
            )
    return resolve_database_url(base_dsn_for_resolve, target_hub)


_settings = get_settings()
_x_args = context.get_x_argument(as_dictionary=False)
_hub_arg = parse_hub_x_arg(list(_x_args))
_resolved_dsn = resolve_env_database_url(
    _settings.database_url, _hub_arg, default_hub=_settings.hub_name
)
config.set_main_option("sqlalchemy.url", _resolved_dsn)
```

# R-V3-3 mitigation (PROJECT.md): Per-hub Alembic drift = MEDIUM risk.
# Mỗi DB hub có alembic_version table riêng — phải khớp head revision sau
# make migrate-all. Script này lint check + exit 1 nếu drift.
#
# Caller chính:
#   - api/Makefile target `migrate-all` (sau loop upgrade head)
#   - ../.github/workflows/test.yml step CI (Plan 05 Task 3)
#
# Yêu cầu prereq:
#   - 4 DB tồn tại (Plan 01-01 init-db.sh)
#   - DATABASE_URL=...medinet_central + HUB_NAME=central trong .env (base DSN)
#   - uv installed + alembic configured

set -euo pipefail

HUBS_ALL=${HUBS_ALL:-"central yte duoc hcns"}
UV=${UV:-uv}

echo "[alembic-head-check] Collect head SHA per hub..."

declare -A HEADS
FAIL=0

for hub in $HUBS_ALL; do
    # `alembic current` output format: "<rev_id> (head)\n" hoặc nhiều dòng nếu multi-head.
    # Grab first 12 chars sau token "head" KHÔNG đáng tin — dùng `alembic heads` + `current` riêng:
    # parse `current` lấy rev_id (token đầu dòng KHÔNG bắt đầu `INFO`).
    raw=$($UV run alembic -x hub="$hub" current 2>&1 || echo "ERROR")
    # Pattern alembic current 1.x output:
    #   <rev> (head)
    # hoặc rỗng nếu chưa apply migration nào.
    # Extract first non-empty line không match log prefix `INFO`/`WARNING`.
    rev=$(echo "$raw" | grep -v '^INFO\|^WARNING\|^Context impl' | awk 'NF{print $1; exit}')
    if [ -z "$rev" ] || [ "$rev" = "ERROR" ]; then
        echo "[alembic-head-check] FAIL: hub=$hub không lấy được current head (raw: $raw)"
        FAIL=1
        continue
    fi
    HEADS[$hub]=$rev
    echo "[alembic-head-check]   hub=$hub head=$rev"
done

if [ $FAIL -eq 1 ]; then
    echo "[alembic-head-check] FAIL — không lấy được head cho ít nhất 1 hub."
    exit 1
fi

# Compare all heads bằng nhau
FIRST_HUB=$(echo "$HUBS_ALL" | awk '{print $1}')
FIRST_HEAD=${HEADS[$FIRST_HUB]}
for hub in $HUBS_ALL; do
    if [ "${HEADS[$hub]}" != "$FIRST_HEAD" ]; then
        echo "[alembic-head-check] FAIL — drift detected: hub=$hub head=${HEADS[$hub]} ≠ $FIRST_HUB head=$FIRST_HEAD"
        echo "[alembic-head-check] R-V3-3 mitigation triggered. Run 'make migrate-all' để re-apply."
        exit 1
    fi
done

echo "[alembic-head-check] PASS — tất cả 4 DB cùng head SHA: $FIRST_HEAD"
exit 0
