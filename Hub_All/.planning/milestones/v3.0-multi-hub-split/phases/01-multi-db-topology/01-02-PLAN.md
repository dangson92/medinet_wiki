```python
from pydantic import model_validator

# Bên dưới class Settings — sau field_validator _no_lan_in_prod
@model_validator(mode="after")
def _enforce_hub_dsn_match(self) -> "Settings":
    """E-V3-3 enforce — hub_name phải khớp database name trong DSN.

    Chống: deploy với HUB_NAME=yte nhưng DATABASE_URL trỏ medinet_hub_duoc
    (hoặc tệ hơn medinet_central) → process truy cập sai data hub. Fail-fast
    ở startup, không defer runtime.

    Quy ước:
    - hub_name == "central"  → database_url phải kết thúc /medinet_central
    - hub_name == "<hub>"    → database_url phải kết thúc /medinet_hub_<hub>

    Cho phép DSN query string (`?option=value`) — strip trước khi check suffix.
    """
    # Strip query string nếu có
    dsn_path = self.database_url.split("?", 1)[0].rstrip("/")
    expected_db = "medinet_central" if self.hub_name == "central" else f"medinet_hub_{self.hub_name}"
    if not dsn_path.endswith(f"/{expected_db}"):
        raise ValueError(
            f"DSN mismatch hub_name: HUB_NAME={self.hub_name!r} yêu cầu database "
            f"{expected_db!r} nhưng DATABASE_URL trỏ {dsn_path.rsplit('/', 1)[-1]!r}. "
            f"E-V3-3 enforce — KHÔNG fallback central."
        )
    return self
```

```python
def resolve_database_url(base_dsn: str, hub_name: str) -> str:
    """Resolve DSN central → DSN hub con bằng cách thay tên database cuối path.

    Dùng cho:
    - `make migrate-all` loop apply Alembic per-hub (Plan 03a sẽ consume qua `-x hub=<name>`)
    - `make hub-init HUB=<name>` dynamic add hub (Plan 04)

    Logic:
    - hub_name == "central" → trả nguyên DSN gốc
    - hub_name == "<hub>"   → thay segment `medinet_central` thành `medinet_hub_<hub>`

    Args:
        base_dsn: DSN trỏ medinet_central (vd "postgresql+asyncpg://u:p@h:5432/medinet_central")
        hub_name: "central" | "yte" | "duoc" | "hcns"

    Returns:
        DSN đã resolve — `medinet_central` thay bằng `medinet_hub_<name>` (nếu hub_name != "central").

    Raises:
        ValueError: base_dsn không kết thúc bằng "/medinet_central" → caller dùng sai input.
    """
    if hub_name == "central":
        return base_dsn
    # Strip query string trước khi thay segment để giữ nguyên query
    if "?" in base_dsn:
        path_part, query_part = base_dsn.split("?", 1)
        query_suffix = f"?{query_part}"
    else:
        path_part, query_suffix = base_dsn, ""
    if not path_part.endswith("/medinet_central"):
        raise ValueError(
            f"base_dsn phải kết thúc bằng '/medinet_central' để resolve "
            f"per-hub; nhận: {path_part!r}"
        )
    new_path = path_part[: -len("medinet_central")] + f"medinet_hub_{hub_name}"
    return new_path + query_suffix
```
