```python
    async def list(
        self,
        *,
        hub_id: UUID | None = None,
        status_filter: str | None = None,
        uploaded_by: UUID | None = None,
        search: str | None = None,
        page: int = 1,
        per_page: int = 20,
    ) -> tuple[list[DocumentListItem], int]:
        """List documents với filter + pagination. Return (items, total).

        Args:
            hub_id: filter theo hub_id (UUID).
            status_filter: filter status enum.
            uploaded_by: filter user upload.
            search: ILIKE search trên filename ('%search%').
            page: 1-based.
            per_page: cap min(per_page, 100) by router (KHÔNG cap ở service).
        """
        # Build WHERE clauses + params
        where_clauses: list[str] = []
        params: dict[str, Any] = {}
        if hub_id is not None:
            where_clauses.append("hub_id = :hub_id")
            params["hub_id"] = str(hub_id)
        if status_filter is not None:
            where_clauses.append("status = :status")
            params["status"] = status_filter
        if uploaded_by is not None:
            where_clauses.append("uploaded_by = :uploaded_by")
            params["uploaded_by"] = str(uploaded_by)
        if search:
            where_clauses.append("filename ILIKE :search")
            params["search"] = f"%{search}%"

        where_sql = ("WHERE " + " AND ".join(where_clauses)) if where_clauses else ""

        # Count
        total_row = (
            await self.db.execute(
                text(f"SELECT COUNT(*) FROM documents {where_sql}"),
                params,
            )
        ).fetchone()
        total = int(total_row[0]) if total_row else 0

        # Page rows
        offset = max(0, (page - 1) * per_page)
        params_with_pg = {**params, "limit": per_page, "offset": offset}
        rows = (
            await self.db.execute(
                text(
                    f"SELECT id, hub_id, filename, status, chunk_count, created_at, updated_at "
                    f"FROM documents {where_sql} "
                    f"ORDER BY created_at DESC "
                    f"LIMIT :limit OFFSET :offset"
                ),
                params_with_pg,
            )
        ).fetchall()
        items = [
            DocumentListItem(
                id=str(r[0]),
                hub_id=str(r[1]),
                filename=r[2],
                status=r[3],
                chunk_count=r[4],
                created_at=r[5],
                updated_at=r[6],
            )
            for r in rows
        ]
        return items, total
```

```python
Mỗi 60s, UPDATE documents SET status='failed' WHERE:
    status='processing'
    AND last_heartbeat IS NOT NULL          -- WARNING #7 fix

Mitigation P8 + WARNING #7:

- WARNING #7 fix combo (b)+(c):
  (b) Plan 04-04 đã bootstrap `last_heartbeat=NOW()` lúc INSERT documents row →
      mọi processing row CÓ initial heartbeat NOT NULL.
  (c) Query CHỈ flip nếu NOT NULL + stale > 2 phút → KHÔNG flip rows mới INSERT
      mà cocoindex worker chưa update heartbeat lần đầu (accept leak ngắn ngủi
      processing nếu chưa có heartbeat update từ flow).

Lifespan integration:
- startup: APPEND-ONLY sau cocoindex.FlowLiveUpdater (Plan 04-03 — WARNING #5):
    app.state.watchdog_task = asyncio.create_task(watchdog_loop())
- shutdown: app.state.watchdog_task.cancel() + await với suppress(CancelledError)

Cocoindex flow Plan 04-03 KHÔNG implement explicit heartbeat update — bootstrap
từ router Plan 04-04 đủ cho M2a. Phase 10 hardening (HARD-04) có thể thêm async
heartbeat tick trong cocoindex flow op để keep heartbeat fresh trong processing
window > 2 phút.
"""
from __future__ import annotations

import asyncio
import logging

from sqlalchemy import text

from app.db.session import get_engine

logger = logging.getLogger(__name__)

WATCHDOG_INTERVAL_SECONDS: int = 60
WATCHDOG_STUCK_THRESHOLD: str = "2 minutes"


async def watchdog_tick() -> int:
    """1 lần tick: flip stuck processing → failed. Return số rows update.

    Idempotent — chạy nhiều lần OK. Test chạy 1 tick → assert count.

    WARNING #7 fix — CHỈ flip nếu `last_heartbeat IS NOT NULL` AND stale.
    Rows processing với last_heartbeat=NULL KHÔNG flip (accept leak).
    """
    try:
        engine = get_engine()
    except RuntimeError:
        logger.warning("watchdog_skip: DB engine chưa init")
        return 0

    async with engine.begin() as conn:
        result = await conn.execute(
            text(
                "UPDATE documents "
                "SET status='failed', "
                "    error_message='timeout: no heartbeat for >120s', "
                "    updated_at=NOW() "
                "WHERE status='processing' "
                "  AND last_heartbeat IS NOT NULL "  # WARNING #7 — NOT NULL guard
                f"  AND last_heartbeat < NOW() - INTERVAL '{WATCHDOG_STUCK_THRESHOLD}'"
            )
        )
        count = result.rowcount or 0

    if count > 0:
        logger.info("watchdog_flipped_stuck_processing: count=%d", count)
    return count


async def watchdog_loop() -> None:
    """Forever-running loop — tick mỗi WATCHDOG_INTERVAL_SECONDS.

    Cancel-safe: nhận CancelledError → return gracefully.
    Exception khác → log + tiếp tục (KHÔNG crash task).
    """
    logger.info("watchdog_loop_start: interval=%ds threshold=%s",
                WATCHDOG_INTERVAL_SECONDS, WATCHDOG_STUCK_THRESHOLD)
    while True:
        try:
            await asyncio.sleep(WATCHDOG_INTERVAL_SECONDS)
            await watchdog_tick()
        except asyncio.CancelledError:
            logger.info("watchdog_loop_cancelled")
            return
        except Exception as e:  # noqa: BLE001 — KHÔNG crash task
            logger.exception("watchdog_tick_failed: %s", e)
            # tiếp tục loop, không break.
```
