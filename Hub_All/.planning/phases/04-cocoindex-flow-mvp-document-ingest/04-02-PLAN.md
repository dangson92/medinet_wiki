```python
def _is_pdf_scanned(reader: PdfReader) -> bool:
    """Heuristic: PDF scanned nếu > 80% pages có < 30 ký tự text non-whitespace.

    pypdf trả empty string hoặc whitespace/garbage cho scanned PDF.
    Threshold 30 ký tự cover edge case: text "Trang 1" + page number cho scanned scan.
    """
    if len(reader.pages) == 0:
        return True
    empty_pages = 0
    for page in reader.pages:
        text = page.extract_text() or ""
        if len(text.strip()) < 30:
            empty_pages += 1
```

```python
"""File extract service — Plan 04-02 (INGEST-02 prerequisite).

Hỗ trợ 4 format: DOCX, TXT, MD, PDF text-only.
KHÔNG OCR (D4 — defer v4.0). Scanned PDF → is_scanned=True (R4 mitigation).
"""
from __future__ import annotations

from pathlib import Path
from typing import Any

import chardet
from docx import Document as DocxDocument
from pypdf import PdfReader

ALLOWED_EXTENSIONS: frozenset[str] = frozenset({".docx", ".txt", ".md", ".pdf"})


class UnsupportedFormatError(Exception):
    """Raise khi extension không nằm trong ALLOWED_EXTENSIONS."""

    def __init__(self, ext: str) -> None:
        super().__init__(
            f"Định dạng {ext!r} không hỗ trợ trong M2. "
            f"Khuyến nghị: chuyển sang một trong {sorted(ALLOWED_EXTENSIONS)}."
        )
        self.ext = ext


def extract_text(file_path: Path) -> tuple[str, bool, dict[str, Any]]:
    """Extract text + scanned-detect + metadata."""
    if not file_path.exists():
        raise FileNotFoundError(f"File không tồn tại: {file_path}")
    ext = file_path.suffix.lower()
    if ext not in ALLOWED_EXTENSIONS:
        raise UnsupportedFormatError(ext)

    if ext == ".docx":
        return _extract_docx(file_path)
    if ext in (".txt", ".md"):
        return _extract_text_file(file_path, ext)
    if ext == ".pdf":
        return _extract_pdf(file_path)
    # Defensive — không reach do guard ALLOWED_EXTENSIONS.
    raise UnsupportedFormatError(ext)


def _extract_docx(file_path: Path) -> tuple[str, bool, dict[str, Any]]:
    doc = DocxDocument(str(file_path))
    paragraphs = [p.text for p in doc.paragraphs if p.text.strip()]
    text = "\n".join(paragraphs)
    meta: dict[str, Any] = {
        "pages": 1,  # DOCX không có page count semantic — set 1 cho monotonic.
        "format": "docx",
        "encoding": "utf-8",
        "paragraph_count": len(paragraphs),
    }
    return text, False, meta


def _extract_text_file(file_path: Path, ext: str) -> tuple[str, bool, dict[str, Any]]:
    raw = file_path.read_bytes()
    # Thử UTF-8 trước (case common cho VN medical doc).
    try:
        text = raw.decode("utf-8")
        encoding = "utf-8"
    except UnicodeDecodeError:
        # Fallback chardet detect (Windows-1258 / latin-1 / etc).
        detected = chardet.detect(raw)
        encoding = detected["encoding"] or "latin-1"
        text = raw.decode(encoding, errors="replace")
    meta: dict[str, Any] = {
        "pages": 1,
        "format": ext.lstrip("."),
        "encoding": encoding,
    }
    return text, False, meta


def _extract_pdf(file_path: Path) -> tuple[str, bool, dict[str, Any]]:
    reader = PdfReader(str(file_path))
    pages_text: list[str] = []
    for page in reader.pages:
        page_text = page.extract_text() or ""
        pages_text.append(page_text)
    text = "\n".join(pages_text)
    is_scanned = _is_pdf_scanned(reader)
    meta: dict[str, Any] = {
        "pages": len(reader.pages),
        "format": "pdf",
        "encoding": "utf-8",  # pypdf decode → str unicode.
    }
    return text, is_scanned, meta


def _is_pdf_scanned(reader: PdfReader) -> bool:
    """Heuristic: PDF scanned nếu > 80% pages có < 30 ký tự text non-whitespace.

    pypdf trả empty/whitespace cho scanned PDF (R4 mitigation).
    Threshold 30 ký tự cover edge case "Trang 1" / page number.
    """
    if len(reader.pages) == 0:
        return True
    empty_pages = 0
    for page in reader.pages:
        text = page.extract_text() or ""
        if len(text.strip()) < 30:
            empty_pages += 1
    return (empty_pages / len(reader.pages)) > 0.8
```

**Strategy (P13 mitigation):**

1. **Pre-process:** Detect heading markers theo regex `^(?P<heading>(Chương|Mục)\s+\d+\.|^\d+\.\s+|^[IVX]+\.\s+)[A-ZĐÁÀẢÃẠÂẦẤẨẪẬĂẰẮẲẴẶÔỒỐỔỖỘƠỜỚỞỠỢƯỪỨỬỮỰÊỀẾỂỄỆ]` (heading + VN caps). Lưu vị trí + heading text.

```python
"""Vietnamese chunker — Plan 04-02 (INGEST-02 prerequisite).

Char-based (P14 cross-provider — OpenAI cl100k_base ≠ Gemini BPE → KHÔNG token-based).
Custom regex VN heading + sentence boundary (P13 mitigation):
- Heading: "Mục N.", "Chương N.", "1.", "2.", "I.", "II." + VN caps.
- Sentence: "." + space + VN caps (Đ Á À Ả ...).

Lý do KHÔNG dùng cocoindex RecursiveSplitter built-in language='vietnamese':
- Cocoindex 1.0.3 KHÔNG có language Vietnamese — fallback default chunker split sai.
- Custom regex pass vào RecursiveSplitter `separators_regex` parameter — Plan 04-04
  wrap chunker này thành cocoindex.op.function để flow gọi.
"""
from __future__ import annotations

import re
from dataclasses import dataclass

# VN caps + ASCII caps (heading first char).
_VN_CAPS = r"A-ZĐÁÀẢÃẠÂẦẤẨẪẬĂẰẮẲẴẶÔỒỐỔỖỘƠỜỚỞỠỢƯỪỨỬỮỰÊỀẾỂỄỆỲÝỶỸỴ"

# Heading patterns — bắt đầu dòng + heading marker + VN caps follow.
HEADING_PATTERNS: tuple[re.Pattern[str], ...] = (
    re.compile(rf"(?m)^(?P<h>Chương\s+\d+\.\s*)[\s{_VN_CAPS}]"),
    re.compile(rf"(?m)^(?P<h>Mục\s+\d+\.\s*)[\s{_VN_CAPS}]"),
    re.compile(rf"(?m)^(?P<h>\d+\.\s+)[{_VN_CAPS}]"),
    re.compile(rf"(?m)^(?P<h>[IVX]+\.\s+)[{_VN_CAPS}]"),
)

# Sentence boundary — "." + whitespace + VN caps follow.
SENTENCE_BOUNDARY = re.compile(rf"\.\s+(?=[{_VN_CAPS}])")


@dataclass(frozen=True)
class ChunkDraft:
    """Output chunker — Plan 04-04 sẽ map thành cocoindex chunks table row."""

    content: str
    heading_path: str | None
    page_start: int
    page_end: int


def chunk_vietnamese(
    text: str,
    chunk_size_chars: int = 1200,
    overlap_chars: int = 120,
) -> list[ChunkDraft]:
    """Split text VN theo heading + sentence boundary. Char-based size."""
    if not text or not text.strip():
        return []
    if chunk_size_chars < 100:
        raise ValueError("chunk_size_chars phải >= 100")
    if overlap_chars < 0 or overlap_chars >= chunk_size_chars:
        raise ValueError("overlap_chars phải 0 <= overlap < chunk_size_chars")

    # Step 1: find heading positions.
    headings: list[tuple[int, str]] = []  # (offset, heading_text)
    for pat in HEADING_PATTERNS:
        for m in pat.finditer(text):
            headings.append((m.start(), m.group("h").strip()))
    headings.sort(key=lambda x: x[0])

    # Step 2: split into segments by heading boundaries.
    segments: list[tuple[str, str | None]] = []  # (content, heading_path)
    if not headings:
        segments.append((text, None))
    else:
        # Pre-content trước heading đầu tiên.
        if headings[0][0] > 0:
            segments.append((text[: headings[0][0]], None))
        for i, (offset, heading) in enumerate(headings):
            end = headings[i + 1][0] if i + 1 < len(headings) else len(text)
            segments.append((text[offset:end], heading))

    # Step 3: within each segment, split nếu > chunk_size_chars theo sentence.
    chunks: list[ChunkDraft] = []
    for seg_content, heading_path in segments:
        if not seg_content.strip():
            continue
        if len(seg_content) <= chunk_size_chars:
            chunks.append(
                ChunkDraft(
                    content=seg_content.strip(),
                    heading_path=heading_path,
                    page_start=_page_at(text, text.find(seg_content)),
                    page_end=_page_at(
                        text, text.find(seg_content) + len(seg_content) - 1
                    ),
                )
            )
            continue
        # Greedy bundle sentence cho đến chunk_size_chars.
        sentences = SENTENCE_BOUNDARY.split(seg_content)
        buffer = ""
        for sent in sentences:
            sent = sent.strip()
            if not sent:
                continue
            # +2 cho ". " separator.
            if len(buffer) + len(sent) + 2 > chunk_size_chars and buffer:
                chunks.append(
                    ChunkDraft(
                        content=buffer.strip(),
                        heading_path=heading_path,
                        page_start=1,
                        page_end=1,
                    )
                )
                # Overlap: lấy `overlap_chars` cuối buffer làm prefix cho chunk kế.
                buffer = buffer[-overlap_chars:] + " " + sent if overlap_chars else sent
            else:
                buffer = (buffer + ". " + sent) if buffer else sent
        if buffer.strip():
            chunks.append(
                ChunkDraft(
                    content=buffer.strip(),
                    heading_path=heading_path,
                    page_start=1,
                    page_end=1,
                )
            )

    return chunks


def _page_at(full_text: str, offset: int) -> int:
    """Đếm form-feed `\\f` (pypdf insert giữa pages) trước offset → page number 1-based."""
    if offset < 0:
        return 1
    return full_text[:offset].count("\f") + 1
```

```python
"""LiteLLM embedding wrapper — Plan 04-02 (INGEST-02 prerequisite).

PIN dimensions=1536 cho cả OpenAI + Gemini (R1 mitigation pgvector HNSW index
2000-dim limit + R7 hot-swap cùng dim không re-embed).

Cocoindex flow Plan 04-04 wrap function này thành cocoindex.op.function để
flow gọi async. Settings hot-swap qua get_settings().rag_embedding_provider /
rag_embedding_model — Plan 07 (ASK-04) UI cho phép admin đổi runtime.

LiteLLM 1.82+ API:
    response = await litellm.aembedding(
        model="text-embedding-3-small",  # hoặc "gemini/embedding-001"
        input=[text],                     # list để batch (M2 chỉ pass 1)
        dimensions=1536,                  # PIN R1
    )
    vector = response.data[0]["embedding"]  # list[float] length 1536
"""
from __future__ import annotations

import logging
from typing import Any

import litellm

from app.config import get_settings

logger = logging.getLogger(__name__)

# PIN dimension cho mọi provider (R1 mitigation).
EMBEDDING_DIM: int = 1536


class EmbedderError(Exception):
    """LiteLLM call fail hoặc response shape sai."""


async def embed_text(text: str, model: str | None = None) -> list[float]:
    """Embed 1 text → vector dim 1536.

    Args:
        text: nội dung chunk (UTF-8 string, < 8000 chars để < 8000 tokens cho mọi tokenizer).
        model: override model name (default settings.rag_embedding_model).

    Returns:
        list[float] length 1536.

    Raises:
        EmbedderError: LiteLLM exception hoặc response sai dim.
        ValueError: text rỗng.
    """
    if not text or not text.strip():
        raise ValueError("text rỗng — không embed được")

    settings = get_settings()
    model_name = model or settings.rag_embedding_model

    try:
        response: Any = await litellm.aembedding(
            model=model_name,
            input=[text],
            dimensions=EMBEDDING_DIM,
        )
    except Exception as e:  # noqa: BLE001 — wrap mọi LiteLLM exception
        logger.error("litellm_embed_failed: model=%s err=%s", model_name, e)
        raise EmbedderError(f"LiteLLM embed fail: {e}") from e

    # LiteLLM 1.82+ response: response.data = [{"embedding": [...], "index": 0}]
    if not response or not getattr(response, "data", None):
        raise EmbedderError(f"LiteLLM response thiếu data: {response!r}")

    vector_raw = response.data[0].get("embedding") if isinstance(
        response.data[0], dict
    ) else response.data[0].embedding
    if not isinstance(vector_raw, list) or len(vector_raw) != EMBEDDING_DIM:
        raise EmbedderError(
            f"Vector dim sai: expected {EMBEDDING_DIM}, got {len(vector_raw) if isinstance(vector_raw, list) else type(vector_raw)}"
        )
    return [float(x) for x in vector_raw]
```

```python
"""File store — local backend mặc định.

Save uploaded file vào `settings.file_store_dir / <uuid>.<ext>`.
GDrive backend defer v4.0 (D3 + R6 sidecar — KHÔNG M2 scope).

Plan 04-04 (router /api/documents/upload) gọi FileStore.save().
Plan 04-02 cocoindex flow extract đọc qua FileStore.load().
"""
from __future__ import annotations

import uuid
from pathlib import Path

from app.config import get_settings


class FileStore:
    """Local filesystem backend — `<file_store_dir>/<uuid>.<ext>`."""

    def __init__(self, base_dir: Path | None = None) -> None:
        self.base_dir = base_dir or get_settings().file_store_dir
        self.base_dir.mkdir(parents=True, exist_ok=True)

    def save(self, content: bytes, original_filename: str) -> Path:
        """Lưu content vào file mới UUID. Return Path tuyệt đối.

        Args:
            content: bytes nội dung file.
            original_filename: filename gốc (chỉ dùng để lấy extension).

        Returns:
            Path đến file đã lưu (UUID4 + ext gốc lowercase).
        """
        ext = Path(original_filename).suffix.lower() or ""
        uid = uuid.uuid4()
        target = self.base_dir / f"{uid}{ext}"
        target.write_bytes(content)
        return target.resolve()

    def load(self, path: Path) -> bytes:
        """Đọc file bytes."""
        return path.read_bytes()

    def delete(self, path: Path) -> bool:
        """Xoá file. Idempotent — return True nếu xoá, False nếu KHÔNG tồn tại."""
        if not path.exists():
            return False
        path.unlink()
        return True
```
