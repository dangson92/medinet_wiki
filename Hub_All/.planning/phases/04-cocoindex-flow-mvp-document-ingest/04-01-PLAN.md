```python
"""Cocoindex setup helper — Plan 04-01 (INGEST-01 prerequisite).

Chạy MỘT LẦN sau alembic upgrade head để cocoindex 1.0.3 tự tạo internal state
tables (schema `cocoindex` — P7 mitigation isolate khỏi schema `public` app data).

Sequence:
    1. cocoindex.init() — đọc env COCOINDEX_DATABASE_URL + APP_NAMESPACE +
       COCOINDEX_DB_SCHEMA, init connection pool nội bộ.
    2. import app.rag.flow (nếu module tồn tại — Plan 04-02 sẽ tạo) để
       @cocoindex.flow_def decorator register flow vào cocoindex registry.
    3. cocoindex.setup_flow() — apply schema delta cho flow đã register.

Plan 04-01 ship setup helper RỖNG (skip flow import nếu chưa có).
Plan 04-02 sau khi tạo app/rag/flow.py: chạy lại `make cocoindex-setup` để
cocoindex tạo bảng `cocoindex.medinet_prod__medinet_wiki_ingest__*`.

Lý do isolation schema:
- Pitfall P7 — cocoindex internal tables (lineage, fingerprint, memo cache)
  KHÔNG trộn schema `public` để Alembic env.py include_object filter
  (Plan 02-03 line 40-55) loại trừ chính xác.

Lý do APP_NAMESPACE cố định:
- Pitfall P2 / R5 — APP_NAMESPACE đổi giữa env = re-index toàn bộ corpus.
  M2 PIN "medinet_prod" mọi env (xem CONVENTIONS.md section 3).
"""
from __future__ import annotations

import logging
import os

from app.config import Settings

logger = logging.getLogger(__name__)


def setup_cocoindex(settings: Settings) -> None:
    """Init cocoindex 1.0.3 + apply flow schema.

    Idempotent: chạy nhiều lần OK (cocoindex tự skip nếu schema đã match).

    Raises:
        RuntimeError: cocoindex.init() fail (Postgres không lên hoặc DSN sai).
    """
    # Cocoindex 1.0.3 đọc env vars qua os.environ tại thời điểm init().
    # Set lại để đảm bảo Settings → env (settings có thể override .env mặc định).
    os.environ.setdefault("COCOINDEX_DATABASE_URL", settings.cocoindex_database_url)
    os.environ.setdefault("APP_NAMESPACE", settings.app_namespace)
    os.environ.setdefault("COCOINDEX_DB_SCHEMA", settings.cocoindex_db_schema)

    import cocoindex

    logger.info(
        "cocoindex_setup_start: namespace=%s schema=%s",
        settings.app_namespace,
        settings.cocoindex_db_schema,
    )

    # 1) Init cocoindex (connection pool + database probe).
    cocoindex.init()
    logger.info("cocoindex.init() OK")

    # 2) Import flow module (Plan 04-02 sẽ tạo app/rag/flow.py).
    #    @cocoindex.flow_def decorator chạy ở import time → register vào registry.
    try:
        from app.rag import flow as _flow  # noqa: F401 — side-effect import

        logger.info("cocoindex_flow_imported: %s", _flow.__name__)
    except ImportError as e:
        logger.warning(
            "cocoindex_flow_skip: flow module chưa tồn tại (Plan 04-02 sẽ tạo) — %s",
            e,
        )

    # 3) Apply schema (idempotent).
    cocoindex.setup_flow()
    logger.info("cocoindex.setup_flow() OK")
```

```python
@pytest.mark.critical
@pytest.mark.integration
def test_phase4_migration_upgrade(
    postgres_container: PostgresContainer,
    alembic_cfg: Config,
) -> None:
    """Alembic upgrade head apply 0001 + 0002 — index ix_documents_status_last_heartbeat tồn tại."""
    # Apply migration head (chạy cả 0001 + 0002).
    asyncio.run(asyncio.to_thread(command.upgrade, alembic_cfg, "head"))

    sync_url = postgres_container.get_connection_url().replace(
        "postgresql+psycopg2://", "postgresql://"
    )
    eng = create_engine(sync_url)
    with eng.connect() as conn:
        # 1) Index composite tồn tại
        rows = conn.execute(
            text(
                "SELECT indexname FROM pg_indexes "
                "WHERE tablename='documents' "
                "ORDER BY indexname"
            )
        ).fetchall()
        index_names = {r[0] for r in rows}
        assert "ix_documents_status_last_heartbeat" in index_names, (
            f"Missing watchdog index — actual: {sorted(index_names)}"
        )

        # 2) Watchdog columns tồn tại đúng kiểu (Phase 2 baseline + Plan 04-01 verify).
        cols = conn.execute(
            text(
                "SELECT column_name, data_type, is_nullable "
                "FROM information_schema.columns "
                "WHERE table_name='documents' "
                "AND column_name IN ('last_heartbeat', 'attempts', 'error_message') "
                "ORDER BY column_name"
            )
        ).fetchall()
        col_map = {r[0]: (r[1], r[2]) for r in cols}
        assert "last_heartbeat" in col_map, f"Missing last_heartbeat: {col_map}"
        assert col_map["last_heartbeat"][0] == "timestamp with time zone"
        assert col_map["last_heartbeat"][1] == "YES"  # nullable
        assert "attempts" in col_map
        assert col_map["attempts"][0] == "integer"
        assert col_map["attempts"][1] == "NO"  # NOT NULL default 0
        assert "error_message" in col_map
        assert col_map["error_message"][0] == "text"

        # 3) Status enum CHECK constraint chứa 'failed_unsupported' (R4 Phase 2)
        check_constraints = conn.execute(
            text(
                "SELECT consrc FROM pg_catalog.pg_constraint pgc "
                "JOIN pg_catalog.pg_class pgcl ON pgcl.oid = pgc.conrelid "
                "WHERE pgcl.relname='documents' AND pgc.contype='c'"
            )
        ).fetchall() if False else []  # consrc deprecated pg12+; use pg_get_constraintdef
        # Dùng pg_get_constraintdef cho compat
        check_defs = conn.execute(
            text(
                "SELECT pg_get_constraintdef(c.oid) FROM pg_constraint c "
                "JOIN pg_class t ON t.oid = c.conrelid "
                "WHERE t.relname='documents' AND c.contype='c'"
            )
        ).fetchall()
        check_text = " ".join(str(d[0]) for d in check_defs)
        assert "failed_unsupported" in check_text, (
            f"Status enum thiếu failed_unsupported: {check_text}"
        )

    eng.dispose()
```
