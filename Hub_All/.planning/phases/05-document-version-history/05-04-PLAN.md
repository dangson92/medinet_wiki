async def _wait_audit_row(
    conn: asyncpg.Connection,
    action: str,
    target_type: str | None = None,
    document_id: str | None = None,
    timeout: float = 3.0,
) -> dict[str, Any] | None:
    """Poll audit_logs WHERE action (+ target_type + document_id) sau BackgroundTask emit.

    Carry forward pattern test_audit_actor_metadata.py:92 + memory project_fastapi_bgtask_commit.

    Args:
        conn: asyncpg Connection.
        action: action string ('document.version.create' | 'document.version.restore').
        target_type: optional filter.
        document_id: optional filter qua payload->>'document_id'.
        timeout: max wait seconds.

    Returns:
        payload dict | None nŠļŅu KH√ĒNG t√¨m thŠļ•y trong timeout.
    """
    elapsed = 0.0
    while elapsed < timeout:
        if document_id:
            row = await conn.fetchrow(
                "SELECT payload FROM audit_logs "
                "WHERE action=$1 AND payload->>'document_id' = $2 "
                "ORDER BY created_at DESC LIMIT 1",
                action, document_id,
            )
        elif target_type:
            row = await conn.fetchrow(
                "SELECT payload FROM audit_logs "
                "WHERE action=$1 AND target_type=$2 "
                "ORDER BY created_at DESC LIMIT 1",
                action, target_type,
            )
        else:
            row = await conn.fetchrow(
                "SELECT payload FROM audit_logs WHERE action=$1 "
                "ORDER BY created_at DESC LIMIT 1",
                action,
            )
        if row:
            return dict(row["payload"]) if isinstance(row["payload"], dict) else row["payload"]
        await asyncio.sleep(0.1)
        elapsed += 0.1
    return None


async def _assert_audit_version_metadata(
    conn: asyncpg.Connection,
    *,
    action: str,
    document_id: str,
    expected_role: str,
    expected_hub_id: str | None,
    expected_version_number: int | None = None,
    expected_change_type: str | None = None,
    expected_restored_to: str | None = None,
