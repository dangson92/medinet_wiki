```python
# api/app/mcp/server.py — module-level singleton (KHÔNG dùng FastAPI DI)
_pool: asyncpg.Pool | None = None
_redis: Any = None


def set_pool(pool: asyncpg.Pool, redis: Any = None) -> None:
    """Gọi từ lifespan FastAPI sau khi db_pool init.
    Tham chiếu: ask.py dùng request.app.state; MCP tool không có Request → dùng singleton.
    """
    global _pool, _redis
```

#### Pattern tool definition — cấu trúc đầu tool (ask.py lines 78–96 + search.py pattern):
```python
mcp = FastMCP("Medinet Wiki MCP Server")

@mcp.tool
async def ask_wiki(query: str, hub_id: str | None = None) -> AskAnswer:
    """Docstring LLM-facing — mô tả ngắn gọn + Args + Returns."""
    pool = _get_pool()
    try:
        user = await authenticate_mcp_request(pool)
    except ValueError as e:
        raise ToolError(str(e))

    # Gọi service trực tiếp (D-04) — KHÔNG self-call HTTP
    ask_svc = AskService(pool=pool, redis=_get_redis())
    cross_hub = hub_id is None
    try:
        if cross_hub:
            outcome = await ask_svc.ask_cross_hub(body=..., user=user)
        else:
            outcome = await ask_svc.ask(body=..., user=user)
    except AskError as e:
        raise ToolError(f"LLM lỗi: {e}")
    except HubIsolationError as e:
        raise ToolError(f"HUB_ACCESS_DENIED: {e}")
    except Exception as e:
        raise ToolError(f"Lỗi nội bộ: {e}")

    # D-14: log usage non-blocking (tham chiếu ask.py lines 131–147)
    asyncio.create_task(
        log_usage_event(
            pool,
            user_id=str(user.user.id),
            hub_id=hub_id,
            model=outcome.usage.model,
            prompt_tokens=outcome.usage.prompt_tokens,
            completion_tokens=outcome.usage.completion_tokens,
            total_tokens=outcome.usage.total_tokens,
            cost_usd=outcome.usage.cost_usd,
            request_id=None,
        )
    )
    return AskAnswer(...)
```

#### Pattern xử lý lỗi — 3 nhánh (ask.py lines 114–129):
```python
# Thứ tự exception handler nhất quán với ask.py:
# 1. ValueError (validation) → ToolError message rõ
```

#### Pattern verify X-API-Key + load User (dependencies.py lines 241–283):
```python
# COPY TRỰC TIẾP từ get_api_key_or_jwt (dependencies.py lines 241–283)
# Thay: HTTPException → raise ValueError (không có FastAPI error context trong MCP)

async def authenticate_mcp_request(pool: asyncpg.Pool) -> UserWithHubs:
    headers = get_http_headers(include={"x-api-key"})
    api_key = headers.get("x-api-key")
    if not api_key:
        raise ValueError("MCP_UNAUTHORIZED: thiếu header X-API-Key")

    from sqlalchemy.ext.asyncio import AsyncSession
    engine = get_engine()
    async with AsyncSession(engine) as db:
        svc = ApiKeyService(db=db)
        principal = await svc.verify_key(api_key)       # tên method canonical: verify_key
        if principal is None:
            raise ValueError("MCP_UNAUTHORIZED: API key không hợp lệ hoặc đã thu hồi")

        # Query created_by — COPY từ dependencies.py lines 253–269
        key_row = (
            await db.execute(
                text("SELECT created_by FROM api_keys WHERE id = :id"),
                {"id": principal["id"]},
            )
        ).fetchone()
        if key_row is None or key_row[0] is None:
            raise ValueError("MCP_UNAUTHORIZED: API key không gắn user hợp lệ")

        stmt = select(User).where(User.id == key_row[0], User.is_active.is_(True))
        user = (await db.execute(stmt)).scalar_one_or_none()
        if user is None:
            raise ValueError("MCP_UNAUTHORIZED: User gắn API key đã bị vô hiệu hoá")

        # Load hub_ids — COPY từ get_current_user_with_hubs (dependencies.py lines 215–218)
        from app.models.auth import UserHub
        stmt_hubs = select(UserHub.hub_id).where(UserHub.user_id == user.id)
        hub_ids = [str(h) for h in (await db.execute(stmt_hubs)).scalars().all()]

        # CRITICAL: commit sau verify_key (ApiKeyService.verify_key ghi last_used_at)
        # Tham chiếu MEMORY: project_fastapi_bgtask_commit.md — commit SAU background writes
        await db.commit()

    return UserWithHubs(user=user, hub_ids=hub_ids)
```

**Nguồn:** `project_fastapi_bgtask_commit.md` (MEMORY.md) + `api/app/auth/dependencies.py` verify_key flow
