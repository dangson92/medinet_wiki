    """Phase 4 Plan 04-06 (SYNC-04 / D-V3-Phase4-C1+C3) — Central checksum scheduler.

    Naive asyncio.sleep loop với cron-like time check (D-V3-Phase4-C3 — KHÔNG
    cần APScheduler dep mới). Lifespan task ở central only (skip hub con).

    Cadence (D-V3-Phase4-C1):
    - Daily 2AM: COUNT(*) per hub con vs central → drift ratio gauge.
    - Hourly: TABLESAMPLE BERNOULLI(1) chunks created last 1h → content_hash diff counter.

    Alerts (defer Phase 7 deploy guide AlertManager config):
    - sync_count_drift > 0.01 sustained 7 days → E-V3-5 trigger STOP
    - sync_hash_drift > 0 sustained 1 hour → Slack alert
    """
    from __future__ import annotations

    import asyncio
    import logging
    import uuid
    from datetime import datetime, timezone
    from typing import TYPE_CHECKING

    import asyncpg

    from app.sync.metrics import SYNC_COUNT_DRIFT, SYNC_HASH_DRIFT

    if TYPE_CHECKING:
        from fastapi import FastAPI

    logger = logging.getLogger(__name__)

    TICK_INTERVAL_SECONDS = 60.0  # check every minute


    async def checksum_scheduler_loop(app: FastAPI) -> None:
        """Main scheduler loop — central-only (D-V3-Phase4-A3 pattern carry forward)."""
        from app.config import get_settings

        settings = get_settings()
        if settings.hub_name != "central":
            logger.info("checksum_scheduler_skip_hub_con: hub_name=%s", settings.hub_name)
            return

        if not settings.checksum_hub_dsns:
            logger.warning(
                "checksum_scheduler_no_hub_dsns: CHECKSUM_HUB_DSNS_JSON chưa set "
                "— scheduler no-op (central deploy lần đầu CHƯA register hub con)"
            )
            # Continue loop — operator có thể set env runtime + restart, scheduler resume

        last_daily_run: datetime | None = None
        last_hourly_run: datetime | None = None

        # Initialize hub_con pools dict — connect lazy (defer first tick)
        hub_pools: dict[str, asyncpg.Pool] = {}

        try:
            while True:
                now = datetime.now(timezone.utc)
                hub_dsns = settings.checksum_hub_dsns  # property re-parse mỗi tick (cheap)

                # Ensure pools (lazy connect)
                for hub_name, dsn in hub_dsns.items():
                    if hub_name not in hub_pools:
                        try:
                            from app.db.dsn import _to_asyncpg_dsn  # W3 fix — shared helper module
                            hub_pools[hub_name] = await asyncpg.create_pool(
                                dsn=_to_asyncpg_dsn(dsn),
                                min_size=1, max_size=2,
                            )
                            logger.info("checksum_pool_ready: hub=%s", hub_name)
                        except Exception as e:  # noqa: BLE001
                            logger.warning("checksum_pool_init_failed: hub=%s err=%s", hub_name, e)

                # Daily tick (2AM UTC)
                if _should_run_daily(now, last_daily_run):
                    logger.info("checksum_daily_tick_start: now=%s", now.isoformat())
                    for hub_name in hub_dsns:
                        try:
                            await _tick_daily_count(app.state.db_pool, hub_pools.get(hub_name), hub_name)
                        except Exception as e:  # noqa: BLE001
                            logger.warning("checksum_daily_hub_failed: hub=%s err=%s", hub_name, e)
                    last_daily_run = now

                # Hourly tick
                if _should_run_hourly(now, last_hourly_run):
                    logger.info("checksum_hourly_tick_start: now=%s", now.isoformat())
                    for hub_name in hub_dsns:
                        try:
                            await _tick_hourly_hash(app.state.db_pool, hub_pools.get(hub_name), hub_name)
                        except Exception as e:  # noqa: BLE001
                            logger.warning("checksum_hourly_hub_failed: hub=%s err=%s", hub_name, e)
                    last_hourly_run = now

                await asyncio.sleep(TICK_INTERVAL_SECONDS)
        except asyncio.CancelledError:
            logger.info("checksum_scheduler_cancelled")
            raise
        finally:
            # Cleanup pools
            for hub_name, pool in hub_pools.items():
                try:
                    await pool.close()
                except Exception as e:  # noqa: BLE001
                    logger.warning("checksum_pool_close_failed: hub=%s err=%s", hub_name, e)


    def _should_run_daily(now: datetime, last_run: datetime | None) -> bool:
        """Run at 2 AM UTC if not already run today."""
        if now.hour != 2:
            return False
        if last_run is None:
            return True
        return last_run.date() < now.date()


    def _should_run_hourly(now: datetime, last_run: datetime | None) -> bool:
        """Run every hour."""
        if last_run is None:
            return True
        return (now - last_run).total_seconds() >= 3600.0


    async def _tick_daily_count(
        central_pool: asyncpg.Pool | None,
        hub_pool: asyncpg.Pool | None,
        hub_name: str,
    ) -> None:
        """Daily count drift per hub_id."""
        if central_pool is None or hub_pool is None:
            return
        # Resolve hub_id UUID từ central.hubs WHERE name = hub_name
        async with central_pool.acquire() as conn:
            hub_id_row = await conn.fetchrow(
                "SELECT id FROM hubs WHERE name = $1", hub_name,
            )
            if hub_id_row is None:
                logger.warning("checksum_daily_hub_not_in_registry: hub=%s", hub_name)
                return
            hub_id = hub_id_row["id"]
            central_count = await conn.fetchval(
                "SELECT COUNT(*) FROM chunks WHERE hub_id = $1", hub_id,
            )
        async with hub_pool.acquire() as conn:
            hub_count = await conn.fetchval("SELECT COUNT(*) FROM chunks")

        # Drift ratio — abs(diff) / max(hub_count, 1) — symmetric
        diff = abs(int(central_count) - int(hub_count))
        ratio = diff / max(int(hub_count), 1)
        SYNC_COUNT_DRIFT.labels(hub_name=hub_name).set(ratio)
        logger.info(
            "checksum_daily_tick: hub=%s central=%d hub_con=%d drift_ratio=%.4f",
            hub_name, central_count, hub_count, ratio,
        )


    async def _tick_hourly_hash(
        central_pool: asyncpg.Pool | None,
        hub_pool: asyncpg.Pool | None,
        hub_name: str,
    ) -> None:
        """Hourly TABLESAMPLE 1% hash diff."""
        if central_pool is None or hub_pool is None:
            return
        # Sample ở hub con (last 1h)
        async with hub_pool.acquire() as conn:
            sample = await conn.fetch("""
                SELECT id, content_hash FROM chunks
                TABLESAMPLE BERNOULLI(1)
                WHERE created_at > NOW() - INTERVAL '1 hour'
            """)
        if not sample:
            return  # empty sample — no recent chunks
        chunk_ids = [row["id"] for row in sample]
        hub_hashes = {row["id"]: bytes(row["content_hash"]) for row in sample}

        # Fetch central
        async with central_pool.acquire() as conn:
            central_rows = await conn.fetch(
                "SELECT id, content_hash FROM chunks WHERE id = ANY($1::uuid[])",
                chunk_ids,
            )
        central_hashes = {row["id"]: bytes(row["content_hash"]) for row in central_rows}

        # Compare
        mismatch_count = 0
        missing_count = 0
        for chunk_id, hub_hash in hub_hashes.items():
            if chunk_id not in central_hashes:
                missing_count += 1
            elif central_hashes[chunk_id] != hub_hash:
                mismatch_count += 1

        if mismatch_count > 0:
            SYNC_HASH_DRIFT.labels(hub_name=hub_name, drift_type="mismatch").inc(mismatch_count)
        if missing_count > 0:
            SYNC_HASH_DRIFT.labels(hub_name=hub_name, drift_type="missing").inc(missing_count)
        logger.info(
            "checksum_hourly_tick: hub=%s sampled=%d mismatch=%d missing=%d",
            hub_name, len(sample), mismatch_count, missing_count,
        )


    __all__ = [
        "TICK_INTERVAL_SECONDS",
        "checksum_scheduler_loop",
    ]

            try:

                # asyncpg execute returns "UPDATE N" — parse N
                replayed_count = int(result.split()[-1]) if result.startswith("UPDATE") else 0
            finally:
                await conn.close()

            # W8 fix (CR Iteration 1) — admin audit trail cho replay action
            # T-04-06-03 Repudiation mitigation reinforced — INSERT audit_logs row

                        await audit_conn.execute(

                               VALUES ($1, $2, $3, $4, $5::jsonb)""",
                            user.id, "sync.replay", "sync_outbox", body.hub_id,
                            json.dumps({"since": body.since.isoformat(), "rows_replayed": replayed_count}),
                        )
            except Exception as audit_exc:  # noqa: BLE001
                logger.warning("sync_replay_audit_log_failed: %s", audit_exc)
        except Exception as exc:
            logger.exception("sync_replay_failed: hub=%s err=%s", body.hub_id, exc)
            return resp.service_unavailable(
                code="REPLAY_FAILED",
                message=f"Replay failed cho hub {body.hub_id!r}: {exc!s}",
            )

        logger.info(
            "sync_replay_completed: hub=%s replayed_count=%d since=%s",
            body.hub_id, replayed_count, body.since.isoformat(),
        )
        return resp.ok(data={
            "hub_id": body.hub_id,
            "replayed_count": replayed_count,
            "since": body.since.isoformat(),
        })
