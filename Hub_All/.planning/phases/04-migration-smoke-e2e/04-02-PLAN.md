# === Helper: login + audit query ===

async def _login_get_token(client: httpx.AsyncClient, email: str, password: str = _TEST_PASSWORD) -> str:
    """POST /api/auth/login + return access_token. Raise AssertionError nếu fail."""
    res = await client.post("/api/auth/login", json={"email": email, "password": password})
    assert res.status_code == 200, f"Login fail {email}: {res.status_code} {res.text}"
    data = res.json()
    assert data.get("success") is True, f"Login envelope fail: {data}"
    return data["data"]["access_token"]


async def _assert_audit_actor_metadata(
    conn: asyncpg.Connection,
    action: str,
    target_type: str,
    expected_role: str | None,
    expected_hub_id: str | None,
    timeout: float = 2.0,
) -> None:
    """Poll audit_logs WHERE action + target_type sau BackgroundTask emit + assert metadata."""
    elapsed = 0.0
    row = None
    while elapsed < timeout:
        row = await conn.fetchrow(
            "SELECT payload FROM audit_logs WHERE action=$1 AND target_type=$2 "
            "ORDER BY created_at DESC LIMIT 1",
            action, target_type,
        )
        if row:
            break
        await asyncio.sleep(0.1)
        elapsed += 0.1

    assert row is not None, f"Audit row not found: action={action} target_type={target_type} timeout={timeout}s"

    payload = row["payload"]
    actual_role = payload.get("actor_role") if isinstance(payload, dict) else None
    actual_hub_id = payload.get("actor_hub_id") if isinstance(payload, dict) else None

    assert actual_role == expected_role, (
        f"Audit actor_role mismatch action={action}: expected={expected_role} got={actual_role}"
    )
    assert actual_hub_id == expected_hub_id, (
        f"Audit actor_hub_id mismatch action={action}: expected={expected_hub_id} got={actual_hub_id}"
    )
