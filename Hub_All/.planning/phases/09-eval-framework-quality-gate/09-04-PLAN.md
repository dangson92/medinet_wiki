"""Cleanup state TRƯỚC mỗi eval run (Phase 9 EVAL-02 — idempotent reset).

Mixed strategy (RESEARCH Component Responsibilities):
1. API DELETE /api/documents/:id (cocoindex auto-tombstone chunks)
2. Postgres TRUNCATE chunks defensive (fallback nếu API stuck)
3. Redis FLUSHDB search cache (Phase 6 SEARCH-04 invalidate)

Idempotent: chạy 2 lần không lỗi.
"""
from __future__ import annotations

import argparse
import asyncio
import sys

import httpx
import psycopg
import redis

# Module path: eval/scripts/cleanup.py → import eval.lib
from eval.lib import APIClient, EvalSettings, _dsn, get_eval_hub_id


async def _cleanup_via_api(settings: EvalSettings, hub_id: str) -> int:
    """List documents eval_hub → DELETE từng cái qua API. Trả count."""
    async with APIClient(settings.backend_url, settings.admin_email, settings.admin_password) as api:
        await api.login()
        # List all documents (cap per_page=100 — REQ INGEST-08)
        deleted = 0
        page = 1
        while True:
            r = await api._request_with_retry(
                "GET",
                f"{settings.backend_url}/api/documents",
                params={"hub_id": hub_id, "page": page, "per_page": 100},
            )
            if r.status_code != 200:
                print(f"  WARN: list documents page {page} HTTP {r.status_code}", file=sys.stderr)
                break
            data = r.json().get("data", {})
            docs = data.get("items", []) or data.get("documents", []) or []
            if not docs:
                break
            for d in docs:
                doc_id = d.get("id")
                if not doc_id:
                    continue
                del_r = await api._request_with_retry(
                    "DELETE",
                    f"{settings.backend_url}/api/documents/{doc_id}",
                )
                if del_r.status_code in (200, 204):
                    deleted += 1
                else:
                    print(f"  WARN: DELETE {doc_id} HTTP {del_r.status_code}", file=sys.stderr)
            # Pagination
            meta = r.json().get("meta", {})
            total_pages = meta.get("total_pages", 1)
            if page >= total_pages:
                break
            page += 1
        return deleted


def _cleanup_postgres_defensive(settings: EvalSettings, hub_id: str) -> int:
    """TRUNCATE/DELETE chunks WHERE hub_id=eval_hub (defensive nếu API stuck).

    Cocoindex sẽ re-detect missing source vs target qua content-hash diff.
    """
    deleted = 0
    with psycopg.connect(_dsn(settings)) as conn, conn.cursor() as cur:
        # Plan A: DELETE qua FK constraint (chunks → documents → hub_id)
        cur.execute(
            "DELETE FROM chunks WHERE document_id IN "
            "(SELECT id FROM documents WHERE hub_id = %s)",
            (hub_id,),
        )
        deleted += cur.rowcount
        cur.execute("DELETE FROM documents WHERE hub_id = %s", (hub_id,))
        deleted += cur.rowcount
        conn.commit()
    return deleted


def _cleanup_redis_cache(settings: EvalSettings) -> int:
    """FLUSHDB Redis search cache + Pub/Sub invalidate (Phase 6 SEARCH-04)."""
    r = redis.from_url(settings.__dict__.get("redis_url", None) or "redis://localhost:6379/0")
    deleted = 0
    # Pattern DEL search:* + hub:*:invalidate (Phase 6 cache scheme)
    for pattern in ("search:*", "hub:*:invalidate", "rate_limit:*"):
        keys = list(r.scan_iter(match=pattern, count=1000))
        if keys:
            r.delete(*keys)
            deleted += len(keys)
    return deleted


async def cleanup(
    skip_api: bool = False,
    skip_postgres: bool = False,
    skip_redis: bool = False,
) -> int:
    settings = EvalSettings()

    try:
        hub_id = await get_eval_hub_id(settings)
    except SystemExit as e:
        print(f"FAIL: {e}", file=sys.stderr)
        return 1

    print(f"Cleanup eval_hub (id={hub_id})")

    if not skip_api:
        try:
            n = await _cleanup_via_api(settings, hub_id)
            print(f"  [API] DELETE {n} document(s) (cocoindex auto-tombstone chunks)")
        except (httpx.RequestError, KeyError) as e:
            print(f"  [API] WARN: {e} — fallback Postgres defensive", file=sys.stderr)
    else:
        print("  [API] skipped")

    if not skip_postgres:
        try:
            n = _cleanup_postgres_defensive(settings, hub_id)
            print(f"  [Postgres] DELETE defensive {n} row(s) (chunks + documents)")
        except psycopg.OperationalError as e:
            print(f"  [Postgres] WARN: {e}", file=sys.stderr)
    else:
        print("  [Postgres] skipped")

    if not skip_redis:
        try:
            n = _cleanup_redis_cache(settings)
            print(f"  [Redis] DEL {n} cache key(s)")
        except redis.RedisError as e:
            print(f"  [Redis] WARN: {e}", file=sys.stderr)
    else:
        print("  [Redis] skipped")

    print("Cleanup done.")
    return 0


def main(argv: list[str] | None = None) -> int:
    p = argparse.ArgumentParser(description="Cleanup eval_hub state idempotent")
    p.add_argument("--skip-api", action="store_true", help="Skip API DELETE step")
    p.add_argument("--skip-postgres", action="store_true", help="Skip Postgres TRUNCATE defensive")
    p.add_argument("--skip-redis", action="store_true", help="Skip Redis FLUSHDB cache")
    args = p.parse_args(argv)

    return asyncio.run(
        cleanup(
            skip_api=args.skip_api,
            skip_postgres=args.skip_postgres,
            skip_redis=args.skip_redis,
        )
    )


if __name__ == "__main__":
    sys.exit(main())

```python
"""Eval orchestrator end-to-end (Phase 9 EVAL-02).

Workflow:
  preflight → cleanup → upload 10 file → poll → search 12 query → metrics → EVAL.md → exit.

CLI:
  python -m eval.run_eval                  # default: real LLM (yêu cầu OPENAI_API_KEY)
  python -m eval.run_eval --mock-embed     # smoke regression mode (KHÔNG gate verdict)
  python -m eval.run_eval --skip-cleanup   # skip cleanup step (nếu vừa cleanup xong)

Env switch (RESEARCH Open Q5):
  EVAL_MOCK_EMBED=1  → mock embedding (CI smoke)
  EVAL_USE_REAL_LLM=1 → real OpenAI/Gemini call (Wave 4 gate verdict)
"""
from __future__ import annotations

import argparse
import asyncio
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from eval.lib import APIClient, EvalSettings, get_eval_hub_id, preflight_check, upload_and_wait
from eval.metrics import compute_latency_percentiles, compute_retrieval_metrics
from eval.report import generate_eval_md, gate_verdict


EVAL_ROOT = Path(__file__).resolve().parent
DATASET_SOURCES = EVAL_ROOT / "dataset" / "sources"
DATASET_SCANNED = EVAL_ROOT / "dataset" / "scanned"
QUERIES_PATH = EVAL_ROOT / "queries.jsonl"


def _load_queries() -> list[dict]:
    if not QUERIES_PATH.exists():
        raise SystemExit(f"queries.jsonl không tồn tại: {QUERIES_PATH}")
    out: list[dict] = []
    for line in QUERIES_PATH.read_text(encoding="utf-8").splitlines():
        line = line.strip()
        if not line:
            continue
        out.append(json.loads(line))
    return out


def _collect_dataset_files() -> list[Path]:
    """8 sources + 2 scanned = 10 file."""
    files: list[Path] = []
    if DATASET_SOURCES.exists():
        files.extend(sorted(DATASET_SOURCES.iterdir()))
    if DATASET_SCANNED.exists():
        files.extend(sorted(DATASET_SCANNED.iterdir()))
    if not files:
        raise SystemExit(f"Dataset trống. Restore từ git: git checkout 0af44f0 -- {EVAL_ROOT}/dataset/")
    return files


def _check_llm_credentials(mock_embed: bool) -> None:
    if mock_embed:
        print("⚠️  MOCK EMBED mode — KHÔNG measure gate verdict thật. Chỉ smoke regression.")
        return
    key = os.getenv("OPENAI_API_KEY", "").strip()
    if not key:
        raise SystemExit(
            "OPENAI_API_KEY chưa set — gate verdict YÊU CẦU real LLM call.\n"
            "Set: export OPENAI_API_KEY=sk-...\n"
            "HOẶC chạy smoke mock: python -m eval.run_eval --mock-embed\n"
            "Tiền điều kiện: tài khoản OpenAI tier paid (RESEARCH Open Q2)."
        )


async def run_eval(
    mock_embed: bool = False,
    skip_cleanup: bool = False,
    top_k: int = 10,
    output_path: Path = EVAL_ROOT / "results.json",
    eval_md_path: Path = EVAL_ROOT / "EVAL.md",
) -> int:
    settings = EvalSettings()
    _check_llm_credentials(mock_embed)

    print(f"[1/8] Preflight check (backend={settings.backend_url})...")
    await preflight_check(settings)

    print("[2/8] Resolve eval_hub_id...")
    hub_id = await get_eval_hub_id(settings)
    print(f"      eval_hub_id = {hub_id}")

    if not skip_cleanup:
        print("[3/8] Cleanup state (API DELETE + Postgres + Redis)...")
        from eval.scripts.cleanup import cleanup as _cleanup
        rc = await _cleanup()
        if rc != 0:
            print("  WARN: cleanup non-zero — continue anyway", file=sys.stderr)
    else:
        print("[3/8] Cleanup SKIPPED (--skip-cleanup).")

    dataset_files = _collect_dataset_files()
    print(f"[4/8] Login admin + upload {len(dataset_files)} file...")

    upload_results: list[dict] = []
    async with APIClient(settings.backend_url, settings.admin_email, settings.admin_password) as api:
        await api.login()

        for fp in dataset_files:
            print(f"      upload: {fp.name}")
            res = await upload_and_wait(
                api,
                fp,
                hub_id,
                timeout_sec=settings.upload_timeout_sec,
                poll_sec=settings.poll_interval_sec,
            )
            upload_results.append(res)
            print(f"        → status={res['status']} chunks={res.get('chunk_count', 0)}")

        # Settle cocoindex
        print("[5/8] Settle 2s cho cocoindex flush LMDB...")
        await asyncio.sleep(2.0)

        # Run queries
        queries = _load_queries()
        print(f"[6/8] Run {len(queries)} query qua POST /api/search top_k={top_k}...")
        per_query_results: list[dict] = []
        latencies: list[float] = []
        for q in queries:
            try:
                results, lat_ms = await api.search(q["query"], hub_id, top_k=top_k)
            except Exception as e:  # noqa: BLE001
                print(f"  q={q.get('id')} search FAIL: {e}", file=sys.stderr)
                results, lat_ms = [], 0.0
            per_query_results.append({"results": results})
            latencies.append(lat_ms)

    print("[7/8] Compute metrics...")
    retrieval_metrics = compute_retrieval_metrics(queries, per_query_results)
    latency_stats = compute_latency_percentiles(latencies)

    upload_summary = {
        "completed": sum(1 for u in upload_results if u["status"] == "completed"),
        "failed_unsupported": sum(1 for u in upload_results if u["status"] == "failed_unsupported"),
        "failed": sum(1 for u in upload_results if u["status"] == "failed"),
        "timeout": sum(1 for u in upload_results if u["status"] == "timeout"),
        "documents": upload_results,
    }

    results = {
        "run_metadata": {
            "timestamp": datetime.now(timezone.utc).isoformat(timespec="seconds"),
            "backend_url": settings.backend_url,
            "embedding_model": os.getenv("EVAL_EMBEDDING_MODEL", "openai/text-embedding-3-large@1536"),
            "llm_model": os.getenv("EVAL_LLM_MODEL", "openai/gpt-4o-mini"),
            "eval_hub_id": hub_id,
            "dataset_size": len(dataset_files),
            "query_count": len(queries),
            "mock_embed": mock_embed,
        },
        "upload_summary": upload_summary,
        "retrieval_metrics": retrieval_metrics,
        "latency": latency_stats,
    }

    print(f"[8/8] Write results.json + EVAL.md...")
    output_path.write_text(json.dumps(results, indent=2, ensure_ascii=False), encoding="utf-8")
    md = generate_eval_md(results)
    eval_md_path.write_text(md, encoding="utf-8")

    top_3 = retrieval_metrics["top_3_hit_rate"]
    v = gate_verdict(top_3)
    badge = "✅ PASS" if v["verdict"] == "PASS" else "❌ FAIL"
    print(
        f"\n{'='*60}\n"
        f"{badge}  top-3 hit rate = {top_3:.3f} ({top_3*100:.1f}%)\n"
        f"Exit code: {v['exit_code']}\n"
        f"Report: {eval_md_path}\n"
        f"Raw: {output_path}\n"
        f"{'='*60}"
    )
    if v.get("trigger_e5"):
        print("🚨 TRIGGER E5 — STOP M2b. Discuss với user (PROJECT.md EXIT criteria).")
    if mock_embed:
        print("ℹ️  Mock-embed mode — verdict KHÔNG đại diện chất lượng thật. Re-run với real LLM cho gate.")

    return v["exit_code"]


def main(argv: list[str] | None = None) -> int:
    p = argparse.ArgumentParser(description="Eval orchestrator Phase 9 EVAL-02")
    p.add_argument("--mock-embed", action="store_true",
                   help="Mock embedding (smoke regression — KHÔNG gate verdict)")
    p.add_argument("--skip-cleanup", action="store_true", help="Skip cleanup step")
    p.add_argument("--top-k", type=int, default=10, help="top_k cho search (default 10)")
    p.add_argument("--output", type=Path, default=EVAL_ROOT / "results.json",
                   help="Output results.json path")
    p.add_argument("--eval-md", type=Path, default=EVAL_ROOT / "EVAL.md",
                   help="Output EVAL.md path")
    args = p.parse_args(argv)

    # Env consistency
    mock = args.mock_embed or os.getenv("EVAL_MOCK_EMBED", "").lower() in ("1", "true", "yes")

    return asyncio.run(
        run_eval(
            mock_embed=mock,
            skip_cleanup=args.skip_cleanup,
            top_k=args.top_k,
            output_path=args.output,
            eval_md_path=args.eval_md,
        )
    )


if __name__ == "__main__":
    sys.exit(main())
```

## 5. Troubleshooting

### `Pre-flight FAIL: backend healthz trả 5xx`
- Stack chưa khởi động: `cd Hub_All && docker compose up -d`
- Check uvicorn log: `docker compose logs python-api | tail -50`
- Tìm `cocoindex_init_failed_fail_fast` → fix cocoindex setup trước.

### `Pre-flight FAIL: hub code='eval' chưa seed`
- Chạy: `psql ... -f Hub_All/eval/scripts/seed_hub.sql`
- Verify: `SELECT id FROM hubs WHERE code='eval';`

### Poll timeout 60s
- Phase 4 race fix delay 0.1s + retry 3 attempts → worst-case ~3.6s overhead.
- File lớn (>10MB DOCX VN) có thể cần `EVAL_UPLOAD_TIMEOUT_SEC=120`.

### Scanned PDF trả 415 — KHÔNG phải lỗi
- R4 mitigation: 2 file `dataset/scanned/*.pdf` expected 415 `failed_unsupported`.
- Đếm vào `upload_summary.failed_unsupported`, KHÔNG vào `failed`.
- Defer OCR Vietnamese v4.0 (D4 PROJECT.md).

### JWT 15 phút hết hạn giữa upload 10 file
- APIClient auto-refresh on 401 (Plan 09-02 Task 1).
- Nếu refresh token hết hạn (7d) → fallback re-login auto.

### `EVAL.md` verdict FAIL < 60%
- TRIGGER E5 — STOP M2b. Discuss với user.
- Options: hybrid BM25, reranker, embedding dim 3072, semantic chunker mới.
- Xem Recommendations section EVAL.md cụ thể.

### Cocoindex memo cache hit incorrect (defer Plan 10)
- Edge case khi đổi prompt KHÔNG re-embed: defer Phase 10 (RESEARCH Open Q5).
