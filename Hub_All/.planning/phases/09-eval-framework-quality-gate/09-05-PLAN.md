"""Fixture eval pipeline smoke (Phase 9 EVAL-04).

Bổ sung fixture cho test_eval_pipeline.py mà KHÔNG override conftest.py Phase 4.
- mock_litellm_embed: monkeypatch litellm.aembedding trả 1536-dim deterministic vector
- eval_hub_seeded: INSERT hubs row code='eval' qua async session, yield hub_id, TRUNCATE sau test
"""
from __future__ import annotations

import hashlib
import uuid
from collections.abc import AsyncIterator

import pytest
import pytest_asyncio
from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession


def _deterministic_embedding(input_text: str, dim: int = 1536) -> list[float]:
    """Sinh vector deterministic từ hash của input — tránh randomness nhưng vẫn unique per query.

    KHÔNG dùng cho gate verdict (vector không có semantic). Chỉ dùng cho smoke regression.
    """
    h = hashlib.sha256(input_text.encode("utf-8")).digest()
    # Lặp hash để đủ dim*4 byte, mỗi 4 byte → 1 float trong [-1, 1]
    out: list[float] = []
    pool = bytearray()
    counter = 0
    while len(out) < dim:
        pool.extend(hashlib.sha256(h + counter.to_bytes(4, "big")).digest())
        counter += 1
        while len(pool) >= 4 and len(out) < dim:
            chunk = pool[:4]
            pool = pool[4:]
            val = int.from_bytes(chunk, "big", signed=False)
            # Normalize về [-1, 1]
            out.append((val / (2**32 - 1)) * 2 - 1)
    return out[:dim]

        inputs = [input] if isinstance(input, str) else list(input)

            "model": model,
            "object": "list",
            "usage": {"prompt_tokens": total_tokens, "total_tokens": total_tokens},
        }

"""Eval pipeline smoke regression (Phase 9 EVAL-04).

Smoke MOCK EMBED < 60s. KHÔNG measure gate verdict — đó là track standalone (make eval-all Wave 4).

Test này verify pipeline reachable:
- upload DOCX → cocoindex flow chunk + embed (mock) → pgvector chunks

CI gate: pytest -m critical api/tests/integration/test_eval_pipeline.py PASS.
"""
from __future__ import annotations

import asyncio
import sys
from pathlib import Path

import pytest
from httpx import AsyncClient

# Import eval lib (Hub_All/eval/ ở repo root)
_EVAL_ROOT = Path(__file__).resolve().parents[3] / "eval"
if str(_EVAL_ROOT.parent) not in sys.path:
    sys.path.insert(0, str(_EVAL_ROOT.parent))

from eval.lib import upload_and_wait  # noqa: E402  # Pattern Plan 09-02


DATASET_SOURCES = _EVAL_ROOT / "dataset" / "sources"
DATASET_SCANNED = _EVAL_ROOT / "dataset" / "scanned"


def _pick_sample_docx() -> Path:
    """Pick file DOCX nhỏ nhất để smoke nhanh."""
    candidates = sorted(DATASET_SOURCES.glob("*.docx"), key=lambda p: p.stat().st_size)
    if not candidates:
        pytest.skip(f"Dataset sources trống tại {DATASET_SOURCES}. Restore từ git 0af44f0.")
    return candidates[0]


def _pick_scanned_pdf() -> Path:
    candidates = sorted(DATASET_SCANNED.glob("*.pdf"))
    if not candidates:
        pytest.skip(f"Dataset scanned trống tại {DATASET_SCANNED}.")
    return candidates[0]


class _AsyncClientAdapter:
    """Adapter để upload_and_wait từ eval.lib (vốn dùng httpx.AsyncClient) chạy được với
    ASGITransport in-process (api/tests/integration).

    Wrap quanh AsyncClient + auth header inline (KHÔNG cần JWT thật — test fixture đã admin)."""

    def __init__(self, client: AsyncClient, base_url: str, token: str):
        self._client = client
        self.base_url = base_url.rstrip("/")
        self._token = token
        self.access_token = token
        self.refresh_token = None

    async def login(self) -> None:
        pass  # no-op: token đã có từ fixture

    async def refresh(self) -> None:
        pass

    async def _request_with_retry(self, method: str, url: str, **kw):
        headers = dict(kw.pop("headers", {}) or {})
        headers["Authorization"] = f"Bearer {self._token}"
        return await self._client.request(method, url, headers=headers, **kw)

    async def upload_document(self, file_path: Path, hub_id: str) -> tuple[int, dict]:
        with file_path.open("rb") as f:
            files = {"file": (file_path.name, f.read(), "application/octet-stream")}
            r = await self._request_with_retry(
                "POST",
                f"{self.base_url}/api/documents/upload",
                files=files,
                data={"hub_id": hub_id},
            )
        return r.status_code, r.json()

    async def get_document(self, doc_id: str) -> dict:
        r = await self._request_with_retry("GET", f"{self.base_url}/api/documents/{doc_id}")
        r.raise_for_status()
        return r.json()["data"]

    async def search(self, query: str, hub_id: str, top_k: int = 10):
        import time
        t0 = time.perf_counter()
        r = await self._request_with_retry(
            "POST",
            f"{self.base_url}/api/search",
            json={"query": query, "hub_ids": [hub_id], "top_k": top_k},
        )
        t1 = time.perf_counter()
        r.raise_for_status()
        return r.json()["data"]["results"], (t1 - t0) * 1000.0


@pytest.mark.critical
@pytest.mark.asyncio
async def test_eval_smoke_upload_search_pipeline(
    app_with_cocoindex: AsyncClient,
    admin_token: str,
    eval_hub_seeded: str,
    mock_litellm_embed: None,
):
    """Smoke: upload 1 DOCX VN sample → search → assert pipeline reachable."""
    sample = _pick_sample_docx()
    hub_id = eval_hub_seeded
    api = _AsyncClientAdapter(app_with_cocoindex, base_url="", token=admin_token)

    # Upload + poll
    res = await upload_and_wait(api, sample, hub_id, timeout_sec=30, poll_sec=0.5)
    assert res["status"] == "completed", f"Upload status: {res['status']} err={res.get('error_message')}"
    assert res.get("chunk_count", 0) > 0, "chunk_count phải > 0 sau cocoindex flow"

    # Search
    results, latency_ms = await api.search("định vị trung tâm", hub_id, top_k=5)
    assert isinstance(results, list), "results phải là list"
    assert len(results) >= 1, "Phải có ≥1 result"
    # M2 D-10: result.title = filename
    for r in results:
        assert "title" in r, f"result thiếu field 'title' (D-10): {r}"
        assert r["title"], "title không được rỗng"
        # hub_isolation E4
        assert r.get("hub_id") == hub_id, f"Result leak hub khác: {r.get('hub_id')} != {hub_id}"

    assert latency_ms < 5000, f"Latency {latency_ms}ms quá lâu cho smoke (mock embed)"

@pytest.mark.critical
@pytest.mark.asyncio
async def test_eval_smoke_hub_isolation_no_leak(
    app_with_cocoindex: AsyncClient,
    admin_token: str,
    eval_hub_seeded: str,
    mock_litellm_embed: None,
    db_session,
):
    """E4 hub isolation: chunk hub-A KHÔNG leak khi search hub-B."""
    from sqlalchemy import text
    import uuid

    sample = _pick_sample_docx()
    hub_a_id = eval_hub_seeded  # đã seed code='eval-smoke'

    # Tạo hub-B
    hub_b_id = str(uuid.uuid4())
    await db_session.execute(
        text(
            "INSERT INTO hubs (id, code, name, subdomain, is_active, created_at) "
            "VALUES (:id, 'eval-smoke-b', 'Eval Hub B', 'eval-smoke-b.medinet.vn', TRUE, NOW())"
        ),
        {"id": hub_b_id},
    )
    await db_session.commit()

    try:
        api = _AsyncClientAdapter(app_with_cocoindex, base_url="", token=admin_token)

        # Upload vào hub-A

            assert r.get("hub_id") != hub_a_id, \
                f"E4 LEAK: chunk hub-A xuất hiện trong search hub-B: {r}"
    finally:
        # Cleanup hub-B
        await db_session.execute(
            text("DELETE FROM hubs WHERE id = :hid"), {"hid": hub_b_id}
        )
        await db_session.commit()
