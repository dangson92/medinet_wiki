import json
from pathlib import Path

src = Path("queries.jsonl")
lines_in = src.read_text(encoding="utf-8").strip().splitlines()
assert len(lines_in) == 12, f"Expected 12 queries, got {len(lines_in)}"

out = []
for line in lines_in:
    q = json.loads(line)
    # Thêm hub_id nếu chưa có (idempotent)

    # Đảm bảo có id (M1 có thể dùng "query_id" hoặc "id")

PY
