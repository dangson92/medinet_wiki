    """Map HTTPException → envelope `{success, data, error, meta}` D6.

    CRITICAL: Plan 03-01 ErrorHandlerMiddleware KHÔNG được catch HTTPException
    (đã có `if isinstance(exc, StarletteHTTPException): raise` pass-through).
    Mọi HTTPException từ dependency/route — bao gồm `MISSING_AUTHORIZATION`

    test_rbac_dependency.py::test_anonymous_returns_401_missing_authorization`
    (Task 05) verify shape envelope cho 401 missing-Bearer scenario.
    """
    detail = exc.detail
    if isinstance(detail, dict) and "code" in detail and "message" in detail:
        code = detail["code"]
        message = detail["message"]
    elif isinstance(detail, str):
        code = "ERROR"
        message = detail
    else:
        code = "ERROR"
        message = str(detail)
    body = {
        "success": False,
        "data": None,
        "error": {"code": code, "message": message},
        "meta": None,
    }
    headers = exc.headers or None
    return StarletteJSONResponse(content=body, status_code=exc.status_code, headers=headers)

```python
def require_role(*roles: str) -> Callable:
    """AUTH-04 — gate endpoint chỉ cho role trong `roles` được pass.

    Usage:
        @router.delete("/admin/something", dependencies=[Depends(require_role("admin"))])
        async def delete(...): ...

    Hoặc inject user:
        @router.put("/...")
        async def edit(user: User = Depends(require_role("admin", "editor"))):
            ...

    Trả 403 FORBIDDEN với envelope shape nếu role không match.
    Trả 401 từ `get_current_user` nếu authentication fail (token sai/missing).
    """
    if not roles:
        raise ValueError("require_role cần ít nhất 1 role")
    allowed = set(roles)

    async def _dependency(user: User = Depends(get_current_user)) -> User:
        if user.role not in allowed:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail={
                    "code": "FORBIDDEN",
                    "message": f"Không đủ quyền — yêu cầu một trong {sorted(allowed)}",
                },
            )
        return user

    return _dependency
```
