/**

import React from 'react';
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { render, screen, fireEvent, waitFor } from '@testing-library/react';
import { MemoryRouter } from 'react-router-dom';

async function renderUserManagement(opts: {
  currentRole?: 'admin' | 'hub_admin' | 'viewer';
  createUserMock?: ReturnType<typeof vi.fn>;
} = {}) {
  vi.resetModules();

  vi.doMock('../../contexts/AuthContext', () => ({
    useAuth: () => ({
      isAuthenticated: true,
      user: {
        user: {
          id: 'u1',
          name: 'Test User',
          email: 't@medinet.vn',
          role: opts.currentRole ?? 'admin',
          status: 'active',
          failed_login_count: 0,
          created_at: '2026-01-01',
          updated_at: '2026-01-01',
        },
        roles: [],
      },
      logout: vi.fn(),
      isLoading: false,
      login: vi.fn(),
      refreshUser: vi.fn(),
    }),
    AuthProvider: ({ children }: { children: React.ReactNode }) => <>{children}</>,
  }));

  const createUserMock = opts.createUserMock ?? vi.fn().mockResolvedValue({ success: true, data: { id: 'new-user' } });
  vi.doMock('../../services/api', () => ({
    api: {
      getHubs: vi.fn().mockResolvedValue({
        success: true,
        data: [{ id: '1', code: 'dmd', name: 'Đỗ Minh Đường', status: 'active' }],
      }),
      listUsers: vi.fn().mockResolvedValue({
        success: true,
        data: [],
        meta: { page: 1, per_page: 10, total: 0, total_pages: 1 },
      }),
      createUser: createUserMock,
      changeUserRole: vi.fn().mockResolvedValue({ success: true }),
    },
  }));

  const { default: UserManagement } = await import('../UserManagement');
  const utils = render(
    <MemoryRouter>
      <UserManagement />
    </MemoryRouter>
  );
  return { ...utils, createUserMock };
}

describe('Phase 3 v3.1 FE-01 — UserManagement form 3 option radio', () => {
  beforeEach(() => {
    vi.resetModules();
  });

  it('render 3 radio option với label tiếng Việt đúng UI-SPEC §7.1', async () => {
    await renderUserManagement();
    // Open Add modal (verify modal trigger button presence — depend UI flow)
    await waitFor(() => {
      // 3 label tiếng Việt option
      expect(screen.queryByText(/Admin toàn hệ thống/)).toBeTruthy();
      expect(screen.queryByText(/Quản lý hub này/)).toBeTruthy();
    });
  });

  it('warning banner chỉ mount khi role admin selected', async () => {
    await renderUserManagement();
    // Trigger open form + click admin radio
    const adminRadio = await screen.findByDisplayValue('admin');
    fireEvent.click(adminRadio);
    await waitFor(() => {
      expect(screen.queryByText(/Quyền cao nhất/)).toBeTruthy();
    });
    // Click viewer → warning unmount
    const viewerRadio = screen.getByDisplayValue('viewer');
    fireEvent.click(viewerRadio);
    await waitFor(() => {
      expect(screen.queryByText(/Quyền cao nhất/)).toBeNull();
    });
  });

  it('BE 403 HUB_ADMIN_REQUIRED → toast tiếng Việt đúng UI-SPEC §7.4', async () => {
    const createUserMock = vi.fn().mockResolvedValue({
      success: false,
      data: null,
      error: { code: 'HUB_ADMIN_REQUIRED', message: 'BE generic message' },
    });
    await renderUserManagement({ createUserMock });
    // Trigger submit form with role='admin'
    // ... fill form + submit
    // Assert toast: "Bạn không có quyền tạo user với role Admin toàn hệ thống. Liên hệ Super Admin."

      expect(errorText).toBeTruthy();

  });
});
