    name: CI — Critical path test (HARD-03)

    on:
      push:

        defaults:
          run:

        steps:
          - name: Checkout
            uses: actions/checkout@v4

          - name: Setup Python 3.12
            uses: actions/setup-python@v5
            with:
              python-version: '3.12'

          - name: Setup uv
            uses: astral-sh/setup-uv@v3

              version: 'latest'
              enable-cache: true
              cache-dependency-glob: |
                Hub_All/api/uv.lock
                Hub_All/api/pyproject.toml

          - name: Mypy strict (critical modules)

                app/services/search_service.py app/services/documents_service.py \
                app/services/ask_service.py app/repositories/hub_isolation.py \

          - name: Pytest critical path with coverage gate
